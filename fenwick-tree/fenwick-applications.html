Classic competitive programming techniques using Fenwick Trees: inversion counting, order statistics, coordinate compression, and offline queries.

1. Counting Inversions

An inversion is a pair (i, j) where i < j but a[i] > a[j]. Count all inversions in O(n log n) using BIT as a frequency array.

Array: [3, 1, 2, 5, 4]

Process left to right. For each element a[i]:
  inversions += (elements already inserted) - query(a[i])
  update(a[i], +1)

Step 1: a[0]=3, inversions += 0 - query(3) = 0 - 0 = 0, update(3, +1)
Step 2: a[1]=1, inversions += 1 - query(1) = 1 - 0 = 1, update(1, +1)
Step 3: a[2]=2, inversions += 2 - query(2) = 2 - 1 = 1, update(2, +1)
Step 4: a[3]=5, inversions += 3 - query(5) = 3 - 3 = 0, update(5, +1)
Step 5: a[4]=4, inversions += 4 - query(4) = 4 - 3 = 1, update(4, +1)

Total inversions = 0 + 1 + 1 + 0 + 1 = 3
Pairs: (3,1), (3,2), (5,4) ✓

long long countInversions(vector<int>& a) {
    int n = a.size();
    int maxVal = *max_element(a.begin(), a.end());
    vector<int> bit(maxVal + 1, 0);

    auto update = [&](int i) {
        for (; i <= maxVal; i += i & -i) bit[i]++;
    };
    auto query = [&](int i) {
        int s = 0;
        for (; i > 0; i -= i & -i) s += bit[i];
        return s;
    };

    long long inv = 0;
    for (int i = 0; i < n; i++) {
        inv += i - query(a[i]);  // elements before i that are > a[i]
        update(a[i]);
    }
    return inv;
}

static long countInversions(int[] a) {
    int maxVal = Arrays.stream(a).max().getAsInt();
    int[] bit = new int[maxVal + 1];

    long inv = 0;
    for (int i = 0; i < a.length; i++) {
        // Query: count of elements <= a[i]
        int cnt = 0;
        for (int j = a[i]; j > 0; j -= j & -j) cnt += bit[j];
        inv += i - cnt;
        // Update: mark a[i] as seen
        for (int j = a[i]; j <= maxVal; j += j & -j) bit[j]++;
    }
    return inv;
}

def count_inversions(a):
    max_val = max(a)
    bit = [0] * (max_val + 1)

    def update(i):
        while i <= max_val:
            bit[i] += 1
            i += i & -i

    def query(i):
        s = 0
        while i > 0:
            s += bit[i]
            i -= i & -i
        return s

    inv = 0
    for i, x in enumerate(a):
        inv += i - query(x)
        update(x)
    return inv

print(count_inversions([3, 1, 2, 5, 4]))  # 3

2. Coordinate Compression

When values are large (up to 10^9) but count is small (up to 10^5), compress values to ranks [1, n] before using BIT.

Original:   [100000, 3, 50000, 7, 1000000]
Sorted:     [3, 7, 50000, 100000, 1000000]
Ranks:      [1, 2, 3, 4, 5]

Mapping:    100000 -> 4
            3      -> 1
            50000  -> 3
            7      -> 2
            1000000-> 5

Compressed: [4, 1, 3, 2, 5]

Now we can use BIT of size 5 instead of 1000000!

// Coordinate compression + inversion count
long long countInversionsLarge(vector<int>& a) {
    int n = a.size();

    // Compress coordinates
    vector<int> sorted_a = a;
    sort(sorted_a.begin(), sorted_a.end());
    sorted_a.erase(unique(sorted_a.begin(), sorted_a.end()), sorted_a.end());

    auto compress = [&](int x) {
        return lower_bound(sorted_a.begin(), sorted_a.end(), x) - sorted_a.begin() + 1;
    };

    int m = sorted_a.size();
    vector<int> bit(m + 1, 0);
    auto update = [&](int i) { for (; i <= m; i += i & -i) bit[i]++; };
    auto query = [&](int i) { int s = 0; for (; i > 0; i -= i & -i) s += bit[i]; return s; };

    long long inv = 0;
    for (int i = 0; i < n; i++) {
        int rank = compress(a[i]);
        inv += i - query(rank);
        update(rank);
    }
    return inv;
}

3. Order Statistics (Kth Smallest)

Find the kth smallest element in a dynamic multiset using BIT + binary search on prefix sums.

Technique: Binary descent on BIT (O(log n) instead of O(log^2 n))

The BIT stores frequencies. We want position p where prefix_sum(p) >= k.

Standard approach: Binary search over positions -> O(log^2 n)
Optimal approach: Bit descent -> O(log n)

Bit Descent Algorithm:
  pos = 0, cur_sum = 0
  for bit_pos = log2(n) down to 0:
    next_pos = pos + (1 << bit_pos)
    if next_pos <= n and cur_sum + BIT[next_pos] < k:
      cur_sum += BIT[next_pos]
      pos = next_pos
  return pos + 1  // 1-indexed answer

struct OrderStatsBIT {
    vector<int> bit;
    int n, LOG;

    OrderStatsBIT(int n) : n(n), bit(n + 1, 0) {
        LOG = __lg(n) + 1;
    }

    void update(int i, int delta) {
        for (; i <= n; i += i & -i) bit[i] += delta;
    }

    // Find kth smallest in O(log n) via bit descent
    int kth(int k) {
        int pos = 0;
        for (int pw = (1 << LOG); pw; pw >>= 1) {
            if (pos + pw <= n && bit[pos + pw] < k) {
                pos += pw;
                k -= bit[pos];
            }
        }
        return pos + 1;
    }
};

// Usage: dynamic multiset
OrderStatsBIT os(100);  // values in [1, 100]
os.update(5, 1);   // insert 5
os.update(3, 1);   // insert 3
os.update(7, 1);   // insert 7
os.kth(2);          // 2nd smallest = 5
os.update(5, -1);  // remove 5
os.kth(2);          // 2nd smallest = 7

class OrderStatsBIT:
    def __init__(self, n):
        self.n = n
        self.bit = [0] * (n + 1)
        self.LOG = n.bit_length()

    def update(self, i, delta):
        while i <= self.n:
            self.bit[i] += delta
            i += i & -i

    def kth(self, k):
        pos = 0
        pw = 1 << self.LOG
        while pw:
            if pos + pw <= self.n and self.bit[pos + pw] < k:
                pos += pw
                k -= self.bit[pos]
            pw >>= 1
        return pos + 1

4. Offline Processing with BIT

Sort queries by some criterion and process them in an order that allows BIT to answer efficiently.

Common offline BIT patterns:
  • Sort by value: Process elements in sorted order, BIT tracks positions
  • Sort by right endpoint: For range queries, process left-to-right
  • CDQ Divide & Conquer: Split problem into subproblems, use BIT for cross contributions
  • Sweep line: Process events left-to-right, BIT tracks active elements
