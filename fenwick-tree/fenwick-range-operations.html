Range Update & Range Query

Extend the basic Fenwick Tree to support range updates (add a value to all elements in [l, r]) and range queries (sum of elements in [l, r]).

Variant 1: Range Update + Point Query

Use a BIT on the difference array. If we add v to range [l, r], we do:

  • update(l, +v) and update(r+1, -v) on the BIT
  • Point query at index i = prefix sum of the difference array = actual value at i
Original:     [0, 0, 0, 0, 0, 0, 0, 0]
Range add +3 to [2, 5]:

Difference:   [0, 0, +3, 0, 0, 0, -3, 0]
                      ^              ^
                    BIT.add(2,+3)  BIT.add(6,-3)

Point query at i = prefix_sum(i):
  query(1) = 0
  query(2) = 0 + 3 = 3
  query(3) = 0 + 3 + 0 = 3
  query(5) = 0 + 3 + 0 + 0 + 0 = 3
  query(6) = 0 + 3 + 0 + 0 + 0 - 3 = 0  ✓

Variant 2: Range Update + Range Query

We need two BITs. The idea: after range updates, we want to compute prefix sum efficiently.

Key Mathematical Derivation:

Let d[i] be the difference array (d[i] = a[i] - a[i-1])
Then: a[i] = sum(d[1..i])

Prefix sum: sum(a[1..p]) = sum_{i=1}^{p} sum_{j=1}^{i} d[j]
           = sum_{j=1}^{p} d[j] * (p - j + 1)
           = (p + 1) * sum_{j=1}^{p} d[j] - sum_{j=1}^{p} d[j] * j

So we maintain two BITs:
  B1[i] stores d[i]        -> prefix sum gives sum(d[1..p])
  B2[i] stores d[i] * i    -> prefix sum gives sum(d[j]*j for j=1..p)

prefix_sum(p) = (p + 1) * B1.query(p) - B2.query(p)

For range_add(l, r, v):
  B1.add(l, v);       B1.add(r+1, -v);
  B2.add(l, v*l);     B2.add(r+1, -v*(r+1));

Code: Range Update + Range Query

#include <bits/stdc++.h>
using namespace std;

struct BIT {
    vector<long long> tree;
    int n;
    BIT(int n) : n(n), tree(n + 1, 0) {}
    void add(int i, long long v) {
        for (; i <= n; i += i & -i) tree[i] += v;
    }
    long long query(int i) {
        long long s = 0;
        for (; i > 0; i -= i & -i) s += tree[i];
        return s;
    }
};

struct RangeTree {
    BIT b1, b2;
    int n;

    RangeTree(int n) : n(n), b1(n), b2(n) {}

    // Add v to all elements in [l, r]
    void range_add(int l, int r, long long v) {
        b1.add(l, v);
        b1.add(r + 1, -v);
        b2.add(l, v * l);
        b2.add(r + 1, -v * (r + 1));
    }

    // Prefix sum [1..p]
    long long prefix_sum(int p) {
        return (long long)(p + 1) * b1.query(p) - b2.query(p);
    }

    // Range sum [l..r]
    long long range_sum(int l, int r) {
        return prefix_sum(r) - prefix_sum(l - 1);
    }
};

int main() {
    RangeTree rt(8);
    rt.range_add(2, 5, 3);  // add 3 to [2..5]
    rt.range_add(4, 7, 2);  // add 2 to [4..7]
    // Array: [0, 3, 3, 5, 5, 2, 2, 0]
    cout << rt.range_sum(1, 8) << endl;  // 20
    cout << rt.range_sum(3, 6) << endl;  // 15
}

class RangeTree {
    long[] b1, b2;
    int n;

    RangeTree(int n) {
        this.n = n;
        b1 = new long[n + 1];
        b2 = new long[n + 1];
    }

    void add(long[] bit, int i, long v) {
        for (; i <= n; i += i & -i) bit[i] += v;
    }

    long query(long[] bit, int i) {
        long s = 0;
        for (; i > 0; i -= i & -i) s += bit[i];
        return s;
    }

    void rangeAdd(int l, int r, long v) {
        add(b1, l, v);       add(b1, r + 1, -v);
        add(b2, l, v * l);   add(b2, r + 1, -v * (r + 1));
    }

    long prefixSum(int p) {
        return (p + 1) * query(b1, p) - query(b2, p);
    }

    long rangeSum(int l, int r) {
        return prefixSum(r) - prefixSum(l - 1);
    }
}

class RangeTree:
    def __init__(self, n):
        self.n = n
        self.b1 = [0] * (n + 1)
        self.b2 = [0] * (n + 1)

    def _add(self, bit, i, v):
        while i <= self.n:
            bit[i] += v
            i += i & -i

    def _query(self, bit, i):
        s = 0
        while i > 0:
            s += bit[i]
            i -= i & -i
        return s

    def range_add(self, l, r, v):
        self._add(self.b1, l, v)
        self._add(self.b1, r + 1, -v)
        self._add(self.b2, l, v * l)
        self._add(self.b2, r + 1, -v * (r + 1))

    def prefix_sum(self, p):
        return (p + 1) * self._query(self.b1, p) - self._query(self.b2, p)

    def range_sum(self, l, r):
        return self.prefix_sum(r) - self.prefix_sum(l - 1)

# Usage
rt = RangeTree(8)
rt.range_add(2, 5, 3)
rt.range_add(4, 7, 2)
print(rt.range_sum(1, 8))  # 20
print(rt.range_sum(3, 6))  # 15
