Fibonacci & Climbing Stairs

The gateway to DP. These simple problems teach the fundamental concepts that apply to all DP problems.

Problem 1: Fibonacci Number

Compute the nth Fibonacci number: F(0)=0, F(1)=1, F(n)=F(n-1)+F(n-2)

All 4 Approaches Compared

   Approach 1: Naive Recursion
   ───────────────────────────
   Time: O(2^n)  |  Space: O(n) stack
   Problem: Exponential! Recomputes same values.
   fib(5) makes 15 calls, fib(50) makes ~10^15 calls!

   Approach 2: Memoization (Top-Down)
   ───────────────────────────────────
   Time: O(n)  |  Space: O(n) memo + O(n) stack
   Solution: Cache results in array/hashmap.

   Approach 3: Tabulation (Bottom-Up)
   ──────────────────────────────────
   Time: O(n)  |  Space: O(n) table
   Solution: Fill table iteratively from base cases.

   Approach 4: Space Optimized
   ───────────────────────────
   Time: O(n)  |  Space: O(1)
   Solution: Only keep last 2 values.

Code: All 4 Approaches

#include <bits/stdc++.h>
using namespace std;

// 1. Naive Recursion: O(2^n) time, O(n) space
int fibNaive(int n) {
    if (n <= 1) return n;
    return fibNaive(n-1) + fibNaive(n-2);
}

// 2. Memoization: O(n) time, O(n) space
int memo[101];
int fibMemo(int n) {
    if (n <= 1) return n;
    if (memo[n] != -1) return memo[n];
    return memo[n] = fibMemo(n-1) + fibMemo(n-2);
}

// 3. Tabulation: O(n) time, O(n) space
int fibTab(int n) {
    vector<int> dp(n+1);
    dp[0] = 0; dp[1] = 1;
    for (int i = 2; i <= n; i++)
        dp[i] = dp[i-1] + dp[i-2];
    return dp[n];
}

// 4. Space Optimized: O(n) time, O(1) space
int fibOpt(int n) {
    if (n <= 1) return n;
    int a = 0, b = 1;
    for (int i = 2; i <= n; i++) {
        int c = a + b;
        a = b; b = c;
    }
    return b;
}

    }

# 1. Naive: O(2^n) time
def fib_naive(n):
    if n <= 1: return n
    return fib_naive(n-1) + fib_naive(n-2)

# 2. Memoization: O(n) time, O(n) space
from functools import lru_cache

@lru_cache(maxsize=None)
def fib_memo(n):
    if n <= 1: return n
    return fib_memo(n-1) + fib_memo(n-2)

# 3. Tabulation: O(n) time, O(n) space
def fib_tab(n):
    dp = [0] * (n + 1)
    dp[1] = 1
    for i in range(2, n + 1):
        dp[i] = dp[i-1] + dp[i-2]
    return dp[n]

# 4. Space Optimized: O(n) time, O(1) space
def fib_opt(n):
    if n <= 1: return n
    a, b = 0, 1
    for _ in range(2, n + 1):
        a, b = b, a + b
    return b

Problem 2: Climbing Stairs

You can climb 1 or 2 steps at a time. How many distinct ways to reach the top of n stairs?

   n=1: [1]                         → 1 way
   n=2: [1,1] [2]                   → 2 ways
   n=3: [1,1,1] [1,2] [2,1]        → 3 ways
   n=4: [1,1,1,1] [1,1,2] [1,2,1]
        [2,1,1] [2,2]              → 5 ways

   Pattern: ways(n) = ways(n-1) + ways(n-2)
   This IS Fibonacci! (shifted by 1)

   Why? From step n, you came from either step n-1 or n-2.

Code: Climbing Stairs

// Climbing Stairs: O(n) time, O(1) space
int climbStairs(int n) {
    if (n <= 2) return n;
    int a = 1, b = 2;
    for (int i = 3; i <= n; i++) {
        int c = a + b;
        a = b; b = c;
    }
    return b;
}

// Generalized: can climb 1, 2, or k steps
int climbStairsK(int n, int k) {
    vector<int> dp(n + 1, 0);
    dp[0] = 1;
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= min(i, k); j++) {
            dp[i] += dp[i - j];
        }
    }
    return dp[n];
}

// Climbing Stairs: O(n) time, O(1) space
static int climbStairs(int n) {

        int c = a + b;
        a = b; b = c;
    }
    return b;
}

// Generalized: can climb 1 to k steps
static int climbStairsK(int n, int k) {
    int[] dp = new int[n + 1];
    dp[0] = 1;
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= Math.min(i, k); j++)
            dp[i] += dp[i - j];
    return dp[n];
}

# Climbing Stairs: O(n) time, O(1) space
def climb_stairs(n):
    if n <= 2: return n
    a, b = 1, 2
    for _ in range(3, n + 1):
        a, b = b, a + b
    return b

# Generalized: can climb 1 to k steps
def climb_stairs_k(n, k):
    dp = [0] * (n + 1)
    dp[0] = 1
    for i in range(1, n + 1):
        for j in range(1, min(i, k) + 1):
            dp[i] += dp[i - j]
    return dp[n]

Complexity Summary

ApproachTimeSpaceNotes
Naive RecursionO(2^n)O(n)Never use in practice
MemoizationO(n)O(n)Easy to write, slight overhead
TabulationO(n)O(n)No stack overflow risk
Space OptimizedO(n)O(1)Best for interviews
Matrix ExponentiationO(log n)O(1)For very large n
  • Key Insight: Fibonacci pattern appears in many DP problems (stairs, tiling, paths)
  • Interview Tip: Start with brute force, optimize to memo, then tabulation, then space-optimized
  • Variants: Min cost stairs, k-step stairs, stairs with cost array
