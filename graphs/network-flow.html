Max Flow / Min Cut

The maximum flow problem asks: given a directed graph with edge capacities, a source s, and a sink t, what is the maximum amount of "flow" we can push from s to t without exceeding any edge's capacity?

Key Concepts

Residual Graph

For each edge (u, v) with capacity c and current flow f:

  • Forward edge: remaining capacity = c - f (can push more flow)
  • Backward edge: capacity = f (can "undo" flow already sent)

Augmenting Path

A path from s to t in the residual graph where every edge has positive remaining capacity. The flow can be increased along this path.

Max-Flow Min-Cut Theorem

The maximum flow from s to t equals the minimum capacity of an s-t cut (a partition of vertices into two sets S and T where s is in S and t is in T, and the cut capacity is the sum of capacities of edges from S to T).

Algorithms Overview

Algorithm Method Time Complexity Notes
Ford-Fulkerson DFS to find augmenting paths O(E * max_flow) May not terminate with irrational capacities
Edmonds-Karp BFS (shortest augmenting path) O(V * E^2) Always terminates, polynomial
Dinic's BFS levels + blocking flow via DFS O(V^2 * E) Fast in practice, O(E*sqrt(V)) on unit graphs

Dinic's Algorithm

Step 1 (BFS): Build level graph using BFS from source. Assign levels to each node (distance from source in residual graph).
Step 2 (DFS): Find blocking flow using DFS. Only traverse edges going to the next level. Use "current arc" optimization to avoid re-exploring dead-end edges.
Step 3: Repeat until BFS cannot reach sink (no augmenting path exists).
Why Dinic's is faster: Each BFS phase increases the shortest path length by at least 1. Since the shortest path can be at most V-1, there are at most V-1 phases. Each phase finds a blocking flow in O(VE) time.

Implementation (Dinic's Algorithm)

#include <bits/stdc++.h>
using namespace std;

struct Edge {
    int to, rev;
    long long cap;
};

class Dinic {
    int n;
    vector<vector<Edge>> graph;
    vector<int> level, iter;

    bool bfs(int s, int t) {
        level.assign(n, -1);
        queue<int> q;
        level[s] = 0;
        q.push(s);
        while (!q.empty()) {
            int v = q.front(); q.pop();
            for (auto& e : graph[v]) {
                if (e.cap > 0 && level[e.to] < 0) {
                    level[e.to] = level[v] + 1;
                    q.push(e.to);
                }
            }
        }
        return level[t] >= 0;
    }

    long long dfs(int v, int t, long long f) {
        if (v == t) return f;
        for (int& i = iter[v]; i < graph[v].size(); i++) {
            Edge& e = graph[v][i];
            if (e.cap > 0 && level[v] < level[e.to]) {
                long long d = dfs(e.to, t, min(f, e.cap));
                if (d > 0) {
                    e.cap -= d;
                    graph[e.to][e.rev].cap += d;
                    return d;
                }
            }
        }
        return 0;
    }

public:
    Dinic(int n) : n(n), graph(n), level(n), iter(n) {}

    void addEdge(int from, int to, long long cap) {
        graph[from].push_back({to, (int)graph[to].size(), cap});
        graph[to].push_back({from, (int)graph[from].size() - 1, 0});
    }

    long long maxflow(int s, int t) {
        long long flow = 0;
        while (bfs(s, t)) {
            iter.assign(n, 0);
            long long d;
            while ((d = dfs(s, t, LLONG_MAX)) > 0)
                flow += d;
        }
        return flow;
    }
};

int main() {
    // 4 nodes: 0=S, 1=A, 2=B, 3=T
    Dinic dinic(4);
    dinic.addEdge(0, 1, 10); // S->A cap 10
    dinic.addEdge(0, 2, 10); // S->B cap 10
    dinic.addEdge(1, 2, 2);  // A->B cap 2
    dinic.addEdge(1, 3, 4);  // A->T cap 4
    dinic.addEdge(2, 3, 10); // B->T cap 10

    cout << dinic.maxflow(0, 3) << endl; // Output: 14
    return 0;
}
import java.util.*;

class Dinic {
    static int[] to, rev, next;
    static long[] cap;
    static int[] head;
    static int edgeCnt = 0;
    static int[] level, iter;

    static void init(int n, int maxEdges) {
        head = new int[n];
        Arrays.fill(head, -1);
        to = new int[maxEdges];
        rev = new int[maxEdges];
        next = new int[maxEdges];
        cap = new long[maxEdges];
        level = new int[n];
        iter = new int[n];
    }

    static void addEdge(int u, int v, long c) {
        to[edgeCnt] = v; cap[edgeCnt] = c;
        rev[edgeCnt] = edgeCnt + 1;
        next[edgeCnt] = head[u]; head[u] = edgeCnt++;

        to[edgeCnt] = u; cap[edgeCnt] = 0;
        rev[edgeCnt] = edgeCnt - 1;
        next[edgeCnt] = head[v]; head[v] = edgeCnt++;
    }

    static boolean bfs(int s, int t, int n) {
        Arrays.fill(level, -1);
        Queue<Integer> q = new LinkedList<>();
        level[s] = 0;
        q.add(s);
        while (!q.isEmpty()) {
            int v = q.poll();
            for (int e = head[v]; e != -1; e = next[e]) {
                if (cap[e] > 0 && level[to[e]] < 0) {
                    level[to[e]] = level[v] + 1;
                    q.add(to[e]);
                }
            }
        }
        return level[t] >= 0;
    }

    static long dfs(int v, int t, long f) {
        if (v == t) return f;
        for (; iter[v] != -1; iter[v] = next[iter[v]]) {
            int e = iter[v];
            if (cap[e] > 0 && level[v] < level[to[e]]) {
                long d = dfs(to[e], t, Math.min(f, cap[e]));
                if (d > 0) {
                    cap[e] -= d;
                    cap[rev[e]] += d;
                    return d;
                }
            }
        }
        return 0;
    }

    static long maxflow(int s, int t, int n) {
        long flow = 0;
        while (bfs(s, t, n)) {
            System.arraycopy(head, 0, iter, 0, n);
            long d;
            while ((d = dfs(s, t, Long.MAX_VALUE)) > 0)
                flow += d;
        }
        return flow;
    }

    public static void main(String[] args) {
        int n = 4;
        init(n, 20); // 10 edges * 2 (forward + backward)

        addEdge(0, 1, 10); // S->A cap 10
        addEdge(0, 2, 10); // S->B cap 10
        addEdge(1, 2, 2);  // A->B cap 2
        addEdge(1, 3, 4);  // A->T cap 4
        addEdge(2, 3, 10); // B->T cap 10

        System.out.println(maxflow(0, 3, n)); // Output: 14
    }
}
from collections import deque

class Dinic:
    def __init__(self, n):
        self.n = n
        self.graph = [[] for _ in range(n)]

    def add_edge(self, u, v, cap):
        # Forward edge
        self.graph[u].append([v, len(self.graph[v]), cap])
        # Backward edge (residual)
        self.graph[v].append([u, len(self.graph[u]) - 1, 0])

    def bfs(self, s, t):
        """Build level graph using BFS"""
        self.level = [-1] * self.n
        self.level[s] = 0
        q = deque([s])
        while q:
            v = q.popleft()
            for to, rev, cap in self.graph[v]:
                if cap > 0 and self.level[to] < 0:
                    self.level[to] = self.level[v] + 1
                    q.append(to)
        return self.level[t] >= 0

    def dfs(self, v, t, f):
        """Find blocking flow using DFS"""
        if v == t:
            return f
        while self.iter[v] < len(self.graph[v]):
            e = self.graph[v][self.iter[v]]
            to, rev, cap = e
            if cap > 0 and self.level[v] < self.level[to]:
                d = self.dfs(to, t, min(f, cap))
                if d > 0:
                    e[2] -= d
                    self.graph[to][rev][2] += d
                    return d
            self.iter[v] += 1
        return 0

    def max_flow(self, s, t):
        flow = 0
        while self.bfs(s, t):
            self.iter = [0] * self.n
            while True:
                d = self.dfs(s, t, float('inf'))
                if d == 0:
                    break
                flow += d
        return flow

# Example: 4 nodes (0=S, 1=A, 2=B, 3=T)
dinic = Dinic(4)
dinic.add_edge(0, 1, 10)  # S->A cap 10
dinic.add_edge(0, 2, 10)  # S->B cap 10
dinic.add_edge(1, 2, 2)   # A->B cap 2
dinic.add_edge(1, 3, 4)   # A->T cap 4
dinic.add_edge(2, 3, 10)  # B->T cap 10

print(dinic.max_flow(0, 3))  # Output: 14

Finding the Min Cut

After running max flow, perform BFS from source in the residual graph (only edges with remaining capacity > 0). The min cut consists of edges (u, v) where u is reachable from source but v is not.

Applications

  • Maximum bipartite matching: Add source connected to left vertices, sink connected to right vertices, all capacities 1.
  • Minimum cut: By max-flow min-cut theorem, equals max flow.
  • Edge-disjoint paths: Max number of edge-disjoint s-t paths = max flow with unit capacities.
  • Project selection: Choose projects to maximize profit considering dependencies.
  • Image segmentation: Min cut separates foreground from background.

Complexity Analysis

Algorithm Time Space
Ford-Fulkerson (DFS) O(E * max_flow) O(V + E)
Edmonds-Karp (BFS) O(V * E^2) O(V + E)
Dinic's O(V^2 * E) O(V + E)
Dinic's (unit capacity) O(E * sqrt(V)) O(V + E)
Practical tip: Dinic's algorithm is the go-to for competitive programming. It is simple to implement and fast enough for most problems. For very large graphs, push-relabel algorithms can be faster but are more complex to code.
