Hamiltonian Path & Cycle

Visit every vertex exactly once. An NP-complete problem solvable with bitmask DP for small n (up to ~20 vertices).

Problem Definition

Hamiltonian Path

  • Visit every vertex exactly once
  • Start and end can be different
  • Path (not a cycle)

Hamiltonian Cycle

  • Visit every vertex exactly once
  • Return to starting vertex
  • Also called Hamiltonian circuit
   Hamiltonian Path exists:         No Hamiltonian Path:

   0 --- 1 --- 2                    0 --- 1
   |           |                    |     |
   3 --- 4 --- 5                    2 --- 3 --- 4
                                          |
   Path: 0→1→2→5→4→3 ✓                   5

                                    (vertex 5 is a dead end from 4)

Bitmask DP Solution

State: dp[mask][i] = can we visit exactly the vertices in mask, ending at vertex i?

Transition: dp[mask][i] = true if there exists j such that:
  - dp[mask ^ (1<<i)][j] is true (mask without vertex i, ending at j)
  - edge(j, i) exists (can go from j to i)

Base: dp[1<<i][i] = true for all i (start at any single vertex)

Answer: Hamiltonian path exists if any dp[(1<<n)-1][i] is true 
   Example: n=4, edges: 0-1, 1-2, 2-3, 0-3

   mask = 0001 (only vertex 0): dp[0001][0] = true
   mask = 0010 (only vertex 1): dp[0010][1] = true
   mask = 0011 (vertices 0,1):  dp[0011][1] = true (0→1)
                                dp[0011][0] = true (1→0)
   mask = 0111 (vertices 0,1,2): dp[0111][2] = true (0→1→2)
   mask = 1111 (all vertices):   dp[1111][3] = true (0→1→2→3) ✓

   Hamiltonian Path: 0 → 1 → 2 → 3

Code: Hamiltonian Path (Bitmask DP)

// Hamiltonian Path: O(2^n * n^2) time, O(2^n * n) space
bool hamiltonianPath(int n, vector<vector<int>>& adj) {
    // dp[mask][i] = can visit vertices in mask ending at i
    vector<vector<bool>> dp(1 << n, vector<bool>(n, false));

    // Base: start at any single vertex
    for (int i = 0; i < n; i++)
        dp[1 << i][i] = true;

    // Fill DP
    for (int mask = 1; mask < (1 << n); mask++) {
        for (int i = 0; i < n; i++) {
            if (!(mask & (1 << i))) continue; // i not in mask
            if (!dp[mask][i]) continue;

            for (int j : adj[i]) {
                if (mask & (1 << j)) continue; // j already visited
                dp[mask | (1 << j)][j] = true;
            }
        }
    }

    // Check if all vertices visited, ending anywhere
    int full = (1 << n) - 1;
    for (int i = 0; i < n; i++)
        if (dp[full][i]) return true;
    return false;
}

// Hamiltonian Cycle: check if can return to start
bool hamiltonianCycle(int n, vector<vector<int>>& adj) {
    vector<vector<bool>> dp(1 << n, vector<bool>(n, false));
    dp[1][0] = true; // fix start at vertex 0

    for (int mask = 1; mask < (1 << n); mask++) {
        for (int i = 0; i < n; i++) {
            if (!(mask & (1 << i)) || !dp[mask][i]) continue;
            for (int j : adj[i]) {
                if (!(mask & (1 << j)))
                    dp[mask | (1 << j)][j] = true;
            }
        }
    }

    int full = (1 << n) - 1;
    for (int i = 0; i < n; i++)
        if (dp[full][i]) // visited all, ended at i
            for (int j : adj[i])
                if (j == 0) return true; // can return to start
    return false;
}

static boolean hamiltonianPath(int n, List<List<Integer>> adj) {
    boolean[][] dp = new boolean[1 << n][n];

    for (int i = 0; i < n; i++)
        dp[1 << i][i] = true;

    for (int mask = 1; mask < (1 << n); mask++) {
        for (int i = 0; i < n; i++) {
            if ((mask & (1 << i)) == 0 || !dp[mask][i]) continue;
            for (int j : adj.get(i)) {
                if ((mask & (1 << j)) == 0)
                    dp[mask | (1 << j)][j] = true;
            }
        }
    }

    int full = (1 << n) - 1;
    for (int i = 0; i < n; i++)
        if (dp[full][i]) return true;
    return false;
}

// Hamiltonian Cycle: fix start vertex, check return edge
static boolean hamiltonianCycle(int n, List<List<Integer>> adj) {
    boolean[][] dp = new boolean[1 << n][n];
    dp[1][0] = true;

    for (int mask = 1; mask < (1 << n); mask++)
        for (int i = 0; i < n; i++) {
            if ((mask & (1 << i)) == 0 || !dp[mask][i]) continue;
            for (int j : adj.get(i))
                if ((mask & (1 << j)) == 0)
                    dp[mask | (1 << j)][j] = true;
        }

    int full = (1 << n) - 1;
    for (int i = 0; i < n; i++)
        if (dp[full][i] && adj.get(i).contains(0))
            return true;
    return false;
}

# Hamiltonian Path: O(2^n * n^2)
def hamiltonian_path(n, adj):
    dp = [[False] * n for _ in range(1 << n)]

    # Base: start at any single vertex
    for i in range(n):
        dp[1 << i][i] = True

    for mask in range(1, 1 << n):
        for i in range(n):
            if not (mask & (1 << i)) or not dp[mask][i]:
                continue
            for j in adj[i]:
                if not (mask & (1 << j)):
                    dp[mask | (1 << j)][j] = True

    full = (1 << n) - 1
    return any(dp[full][i] for i in range(n))

# Hamiltonian Cycle
def hamiltonian_cycle(n, adj):
    dp = [[False] * n for _ in range(1 << n)]
    dp[1][0] = True  # fix start at 0

    for mask in range(1, 1 << n):
        for i in range(n):
            if not (mask & (1 << i)) or not dp[mask][i]:
                continue
            for j in adj[i]:
                if not (mask & (1 << j)):
                    dp[mask | (1 << j)][j] = True

    full = (1 << n) - 1
    return any(dp[full][i] and 0 in adj[i] for i in range(n))

TSP (Travelling Salesman Problem)

Find the minimum cost Hamiltonian cycle in a weighted graph.

// TSP: O(2^n * n^2) time, O(2^n * n) space
int tsp(int n, vector<vector<int>>& cost) {
    const int INF = 1e9;
    vector<vector<int>> dp(1 << n, vector<int>(n, INF));
    dp[1][0] = 0; // start at vertex 0

    for (int mask = 1; mask < (1 << n); mask++) {
        for (int i = 0; i < n; i++) {
            if (!(mask & (1 << i)) || dp[mask][i] == INF) continue;
            for (int j = 0; j < n; j++) {
                if (mask & (1 << j)) continue;
                if (cost[i][j] == INF) continue;
                int nmask = mask | (1 << j);
                dp[nmask][j] = min(dp[nmask][j], dp[mask][i] + cost[i][j]);
            }
        }
    }

    // Return to start
    int full = (1 << n) - 1;
    int ans = INF;
    for (int i = 1; i < n; i++)
        if (cost[i][0] != INF)
            ans = min(ans, dp[full][i] + cost[i][0]);
    return ans;
}

// TSP: O(2^n * n^2)
static int tsp(int n, int[][] cost) {
    int INF = (int) 1e9;
    int[][] dp = new int[1 << n][n];
    for (int[] row : dp) Arrays.fill(row, INF);
    dp[1][0] = 0;

    for (int mask = 1; mask < (1 << n); mask++)
        for (int i = 0; i < n; i++) {
            if ((mask & (1 << i)) == 0 || dp[mask][i] == INF) continue;
            for (int j = 0; j < n; j++) {
                if ((mask & (1 << j)) != 0 || cost[i][j] == INF) continue;
                int nmask = mask | (1 << j);
                dp[nmask][j] = Math.min(dp[nmask][j], dp[mask][i] + cost[i][j]);
            }
        }

    int full = (1 << n) - 1, ans = INF;
    for (int i = 1; i < n; i++)
        if (cost[i][0] != INF)
            ans = Math.min(ans, dp[full][i] + cost[i][0]);
    return ans;
}

# TSP: O(2^n * n^2)
def tsp(n, cost):
    INF = float('inf')
    dp = [[INF] * n for _ in range(1 << n)]
    dp[1][0] = 0

    for mask in range(1, 1 << n):
        for i in range(n):
            if not (mask & (1 << i)) or dp[mask][i] == INF:
                continue
            for j in range(n):
                if (mask & (1 << j)) or cost[i][j] == INF:
                    continue
                nmask = mask | (1 << j)
                dp[nmask][j] = min(dp[nmask][j], dp[mask][i] + cost[i][j])

    full = (1 << n) - 1
    return min(
        (dp[full][i] + cost[i][0] for i in range(1, n) if cost[i][0] != INF),
        default=INF
    )

Complexity

  • Time: O(2^n * n^2) — exponential but feasible for n ≤ 20
  • Space: O(2^n * n)
  • Brute Force: O(n!) — much worse for n > 12
  • NP-complete: No known polynomial algorithm exists
