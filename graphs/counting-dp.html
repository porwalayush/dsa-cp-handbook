Counting DP (Catalan, Combinatorics)

Use DP to count combinatorial structures: Catalan numbers (parentheses, BSTs), derangements, Stirling numbers, Bell numbers, and grid paths with obstacles. These problems share a pattern of building counts from smaller sub-problems.

Catalan Numbers

Definition: C(n) counts structures that can be recursively decomposed by a "split point":

Recurrence: C(n) = sum of C(i) * C(n-1-i) for i = 0 to n-1

Closed form: C(n) = (2n)! / ((n+1)! * n!) = C(2n, n) / (n+1)

Base case: C(0) = 1

First values: 1, 1, 2, 5, 14, 42, 132, 429, 1430, ... 
Catalan Number C(n) counts:

1. Valid parenthesizations with n pairs:
   n=3: ((())), (())(), ()(()),  ()()(), (()())  → C(3) = 5

2. Number of BSTs with n nodes:
   n=3: root at 1,2,3 with subtrees being BSTs of remaining
        C(0)*C(2) + C(1)*C(1) + C(2)*C(0) = 2+1+2 = 5

3. Triangulations of a polygon with n+2 sides: C(n)

4. Paths from (0,0) to (n,n) staying below diagonal: C(n)

5. Full binary trees with n+1 leaves: C(n)

Recurrence visualization for C(4):
  C(4) = C(0)*C(3) + C(1)*C(2) + C(2)*C(1) + C(3)*C(0)
       = 1*5      + 1*2      + 2*1      + 5*1
       = 5 + 2 + 2 + 5 = 14

Derangements

Definition: D(n) = number of permutations of n elements with NO fixed points (no element in its original position).

Recurrence: D(n) = (n-1) * (D(n-1) + D(n-2))

Intuition: Element 1 goes to position k (n-1 choices). Then either:
- k goes to position 1 (derangement of remaining n-2): contributes D(n-2)
- k does NOT go to position 1 (derangement of n-1 elements with k "banned" from pos 1): contributes D(n-1)

Base cases: D(0) = 1, D(1) = 0
First values: 1, 0, 1, 2, 9, 44, 265, ...

Stirling Numbers of the Second Kind

Definition: S(n, k) = number of ways to partition n elements into exactly k non-empty subsets.

Recurrence: S(n, k) = k * S(n-1, k) + S(n-1, k-1)

Intuition: For the n-th element, either:
- It goes into one of the k existing subsets: k * S(n-1, k)
- It forms a new subset by itself: S(n-1, k-1)

Base cases: S(0, 0) = 1, S(n, 0) = 0 for n > 0, S(0, k) = 0 for k > 0

Bell Numbers

Definition: B(n) = total number of partitions of n elements = sum of S(n, k) for k = 0 to n.

Recurrence (Bell triangle): B(n) = sum of C(n-1, k) * B(k) for k = 0 to n-1

First values: 1, 1, 2, 5, 15, 52, 203, 877, ...

Grid Counting with Obstacles

Problem: Count paths from (0,0) to (m,n) moving only right or down, avoiding obstacle cells.

State: dp[i][j] = number of paths from (0,0) to (i,j)

Transition:
- If (i,j) is blocked: dp[i][j] = 0
- Otherwise: dp[i][j] = dp[i-1][j] + dp[i][j-1]

Base case: dp[0][0] = 1 (if not blocked)

Without obstacles: C(m+n, m) = (m+n)! / (m! * n!) 
Grid 4x4 with obstacles (X):

  dp values:           Grid:
  1  1  1  1           .  .  .  .
  1  1  0  1           .  .  X  .
  1  2  0  1           .  .  X  .
  1  3  3  4           .  .  .  .

  Answer: dp[3][3] = 4 paths from top-left to bottom-right

  Without obstacles: C(6,3) = 20 paths
  With 2 obstacles: only 4 valid paths remain

// Counting DP: Catalan, Derangements, Stirling, Bell, Grid Paths
const int MOD = 1e9 + 7;

// Catalan Numbers - O(n^2) DP version
vector<long long> catalan(int n) {
    vector<long long> C(n + 1, 0);
    C[0] = 1;
    for (int i = 1; i <= n; i++)
        for (int j = 0; j < i; j++)
            C[i] = (C[i] + C[j] * C[i-1-j]) % MOD;
    return C;
}

// Derangements - O(n) time, O(n) space
vector<long long> derangements(int n) {
    vector<long long> D(n + 1, 0);
    D[0] = 1; // empty permutation
    if (n >= 1) D[1] = 0;
    for (int i = 2; i <= n; i++)
        D[i] = (i - 1) * (D[i-1] + D[i-2]) % MOD;
    return D;
}

// Stirling Numbers S(n,k) - O(n*k) time and space
vector<vector<long long>> stirling(int n, int k) {
    vector<vector<long long>> S(n + 1, vector<long long>(k + 1, 0));
    S[0][0] = 1;
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= min(i, k); j++)
            S[i][j] = (j * S[i-1][j] + S[i-1][j-1]) % MOD;
    return S;
}

// Bell Numbers - O(n^2) time and space
vector<long long> bell(int n) {
    auto S = stirling(n, n);
    vector<long long> B(n + 1, 0);
    for (int i = 0; i <= n; i++)
        for (int j = 0; j <= i; j++)
            B[i] = (B[i] + S[i][j]) % MOD;
    return B;
}

// Grid paths with obstacles - O(m*n) time and space
long long gridPaths(vector<vector<int>>& grid) {
    int m = grid.size(), n = grid[0].size();
    vector<vector<long long>> dp(m, vector<long long>(n, 0));

    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            if (grid[i][j] == 1) { dp[i][j] = 0; continue; }
            if (i == 0 && j == 0) { dp[i][j] = 1; continue; }
            if (i > 0) dp[i][j] += dp[i-1][j];
            if (j > 0) dp[i][j] += dp[i][j-1];
            dp[i][j] %= MOD;
        }
    }
    return dp[m-1][n-1];
}

// Counting DP: Catalan, Derangements, Stirling, Bell, Grid Paths
static final long MOD = 1_000_000_007;

// Catalan Numbers - O(n^2)
static long[] catalan(int n) {
    long[] C = new long[n + 1];
    C[0] = 1;
    for (int i = 1; i <= n; i++)
        for (int j = 0; j < i; j++)
            C[i] = (C[i] + C[j] * C[i-1-j]) % MOD;
    return C;
}

// Derangements - O(n)
static long[] derangements(int n) {
    long[] D = new long[n + 1];
    D[0] = 1;
    if (n >= 1) D[1] = 0;
    for (int i = 2; i <= n; i++)
        D[i] = (i - 1) * ((D[i-1] + D[i-2]) % MOD) % MOD;
    return D;
}

// Stirling Numbers S(n,k) - O(n*k)
static long[][] stirling(int n, int k) {
    long[][] S = new long[n + 1][k + 1];
    S[0][0] = 1;
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= Math.min(i, k); j++)
            S[i][j] = (j * S[i-1][j] + S[i-1][j-1]) % MOD;
    return S;
}

// Bell Numbers - O(n^2)
static long[] bell(int n) {
    long[][] S = stirling(n, n);
    long[] B = new long[n + 1];
    for (int i = 0; i <= n; i++)
        for (int j = 0; j <= i; j++)
            B[i] = (B[i] + S[i][j]) % MOD;
    return B;
}

// Grid paths with obstacles - O(m*n)
static long gridPaths(int[][] grid) {
    int m = grid.length, n = grid[0].length;
    long[][] dp = new long[m][n];

    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            if (grid[i][j] == 1) { dp[i][j] = 0; continue; }
            if (i == 0 && j == 0) { dp[i][j] = 1; continue; }
            if (i > 0) dp[i][j] += dp[i-1][j];
            if (j > 0) dp[i][j] += dp[i][j-1];
            dp[i][j] %= MOD;
        }
    }
    return dp[m-1][n-1];
}

# Counting DP: Catalan, Derangements, Stirling, Bell, Grid Paths
MOD = 10**9 + 7

# Catalan Numbers - O(n^2) time, O(n) space
def catalan(n):
    C = [0] * (n + 1)
    C[0] = 1
    for i in range(1, n + 1):
        for j in range(i):
            C[i] = (C[i] + C[j] * C[i-1-j]) % MOD
    return C

# Derangements - O(n) time, O(n) space
def derangements(n):
    D = [0] * (n + 1)
    D[0] = 1
    if n >= 1:
        D[1] = 0
    for i in range(2, n + 1):
        D[i] = (i - 1) * (D[i-1] + D[i-2]) % MOD
    return D

# Stirling Numbers S(n,k) - O(n*k) time and space
def stirling(n, k):
    S = [[0] * (k + 1) for _ in range(n + 1)]
    S[0][0] = 1
    for i in range(1, n + 1):
        for j in range(1, min(i, k) + 1):
            S[i][j] = (j * S[i-1][j] + S[i-1][j-1]) % MOD
    return S

# Bell Numbers - O(n^2) time and space
def bell(n):
    S = stirling(n, n)
    B = [0] * (n + 1)
    for i in range(n + 1):
        B[i] = sum(S[i]) % MOD
    return B

# Grid paths with obstacles - O(m*n) time and space
def grid_paths(grid):
    m, n = len(grid), len(grid[0])
    dp = [[0] * n for _ in range(m)]

    for i in range(m):
        for j in range(n):
            if grid[i][j] == 1:
                dp[i][j] = 0
                continue
            if i == 0 and j == 0:
                dp[i][j] = 1
                continue
            if i > 0:
                dp[i][j] += dp[i-1][j]
            if j > 0:
                dp[i][j] += dp[i][j-1]
            dp[i][j] %= MOD

    return dp[m-1][n-1]

# Examples
print(catalan(5))        # [1, 1, 2, 5, 14, 42]
print(derangements(5))   # [1, 0, 1, 2, 9, 44]
print(bell(5))           # [1, 1, 2, 5, 15, 52]

Key Insights

  • Catalan pattern: Look for problems where you "choose a split point" and count structures on both sides independently. The product of sub-counts gives the total for that split
  • Modular arithmetic: For large n, use modular inverse for closed-form Catalan: C(n) = C(2n,n)/(n+1) mod p
  • Derangements in probability: P(derangement) = D(n)/n! approaches 1/e as n grows
  • Grid paths advanced: With multiple obstacles at known positions, use inclusion-exclusion on obstacles or LGV lemma for non-intersecting paths
  • Stirling to Bell: Bell numbers are the "row sums" of Stirling numbers, connecting partition-into-k to total-partitions
