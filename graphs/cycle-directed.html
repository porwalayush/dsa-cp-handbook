Cycle Detection in Directed Graph

In directed graphs, we can't just check parent. We need a smarter approach: 3-color DFS (White-Gray-Black).

Why Parent Check Doesn't Work

  This is NOT a cycle in directed graph:
    0 → 1
    ↓   ↓
    2 → 3     (all arrows go forward, no way to loop back)

  This IS a cycle:
    0 → 1
    ↑   ↓
    3 ← 2     (0→1→2→3→0 forms a cycle)

Method: DFS 3-Coloring

Color each node:

  • WHITE (0): Not yet visited
  • GRAY (1): Currently being processed (in the current DFS path)
  • BLACK (2): Completely processed (all descendants explored)

Rule: If we encounter a GRAY node → CYCLE! (Because gray means it's an ancestor in current path)

  Graph: 0→1→2→3→1 (cycle: 1→2→3→1)

  DFS from 0:
  Color 0 = GRAY
    Color 1 = GRAY
      Color 2 = GRAY
        Color 3 = GRAY
          Neighbor 1 is GRAY → CYCLE FOUND!

  Gray node = "I'm still exploring this path"
  If I meet a gray node again = I came back to where I started = CYCLE

import java.util.*;

public class CycleDirected {
    static final int WHITE = 0, GRAY = 1, BLACK = 2;

    static boolean hasCycleDFS(int node, List<List<Integer>> adj, int[] color) {
        color[node] = GRAY; // mark as "in progress"

        for (int neighbor : adj.get(node)) {
            if (color[neighbor] == GRAY) {
                return true; // found a back edge → CYCLE!
            }
            if (color[neighbor] == WHITE) {
                if (hasCycleDFS(neighbor, adj, color))
                    return true;
            }
            // BLACK nodes are already fully processed, skip them
        }

        color[node] = BLACK; // done with this node
        return false;
    }

    static boolean hasCycle(List<List<Integer>> adj, int V) {
        int[] color = new int[V]; // all WHITE initially

        for (int i = 0; i < V; i++) {
            if (color[i] == WHITE) {
                if (hasCycleDFS(i, adj, color))
                    return true;
            }
        }
        return false;
    }

    public static void main(String[] args) {
        int V = 4;
        List<List<Integer>> adj = new ArrayList<>();
        for (int i = 0; i < V; i++) adj.add(new ArrayList<>());

        // 0→1→2→3→1 (directed cycle)
        adj.get(0).add(1);
        adj.get(1).add(2);
        adj.get(2).add(3);
        adj.get(3).add(1); // back edge creating cycle

        System.out.println("Has cycle: " + hasCycle(adj, V)); // true
    }
}

#include <bits/stdc++.h>
using namespace std;

const int WHITE = 0, GRAY = 1, BLACK = 2;

bool hasCycleDFS(int node, vector<vector<int>>& adj, vector<int>& color) {
    color[node] = GRAY;

    for (int neighbor : adj[node]) {
        if (color[neighbor] == GRAY) return true; // cycle!
        if (color[neighbor] == WHITE) {
            if (hasCycleDFS(neighbor, adj, color))
                return true;
        }
    }

    color[node] = BLACK;
    return false;
}

bool hasCycle(vector<vector<int>>& adj, int V) {
    vector<int> color(V, WHITE);
    for (int i = 0; i < V; i++) {
        if (color[i] == WHITE && hasCycleDFS(i, adj, color))
            return true;
    }
    return false;
}

WHITE, GRAY, BLACK = 0, 1, 2

def has_cycle_dfs(node, adj, color):
    color[node] = GRAY

    for neighbor in adj[node]:
        if color[neighbor] == GRAY:
            return True  # back edge = cycle!
        if color[neighbor] == WHITE:
            if has_cycle_dfs(neighbor, adj, color):
                return True

    color[node] = BLACK
    return False

def has_cycle(adj, V):
    color = [WHITE] * V
    for i in range(V):
        if color[i] == WHITE:
            if has_cycle_dfs(i, adj, color):
                return True
    return False

Alternative: Using Kahn's Algorithm (BFS)

If topological sort doesn't include all nodes → there's a cycle!

static boolean hasCycleKahn(List<List<Integer>> adj, int V) {
    int[] indegree = new int[V];
    for (int i = 0; i < V; i++)
        for (int nb : adj.get(i))
            indegree[nb]++;

    Queue<Integer> q = new LinkedList<>();
    for (int i = 0; i < V; i++)
        if (indegree[i] == 0) q.add(i);

    int count = 0; // nodes processed
    while (!q.isEmpty()) {
        int node = q.poll();
        count++;
        for (int nb : adj.get(node)) {
            indegree[nb]--;
            if (indegree[nb] == 0) q.add(nb);
        }
    }

    // If not all nodes processed → cycle exists!
    return count != V;
}

bool hasCycleKahn(vector<vector<int>>& adj, int V) {
    vector<int> indegree(V, 0);
    for (int i = 0; i < V; i++)
        for (int nb : adj[i])
            indegree[nb]++;

    queue<int> q;
    for (int i = 0; i < V; i++)
        if (indegree[i] == 0) q.push(i);

    int count = 0;
    while (!q.empty()) {
        int node = q.front(); q.pop();
        count++;
        for (int nb : adj[node]) {
            if (--indegree[nb] == 0) q.push(nb);
        }
    }
    return count != V; // cycle if not all processed
}

from collections import deque

def has_cycle_kahn(adj, V):
    indegree = [0] * V
    for i in range(V):
        for nb in adj[i]:
            indegree[nb] += 1

    queue = deque(i for i in range(V) if indegree[i] == 0)
    count = 0

    while queue:
        node = queue.popleft()
        count += 1
        for nb in adj[node]:
            indegree[nb] -= 1
            if indegree[nb] == 0:
                queue.append(nb)

    return count != V  # cycle if not all processed

Summary

  • DFS 3-Coloring: Most intuitive. Gray = currently in path, finding gray = back edge = cycle
  • Kahn's BFS: If topo sort fails (count < V) → cycle exists
  • Key difference from undirected: Can't use parent check! Must use coloring.
