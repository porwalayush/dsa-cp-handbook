0-1 BFS

0-1 BFS is a shortest path algorithm optimized for graphs where all edge weights are either 0 or 1. It uses a deque instead of a priority queue, achieving O(V+E) time complexity compared to Dijkstra's O((V+E) log V).

When to Use

  • All edge weights are exactly 0 or 1.
  • Finding minimum cost paths in grids where some moves are "free" and others cost 1.
  • Minimum flips/changes to reach a destination.
  • Any problem reducible to a 0-1 weighted graph.

Key Idea

Instead of a priority queue (min-heap), we use a deque (double-ended queue):

  • When traversing a 0-weight edge: push the neighbor to the front of the deque.
  • When traversing a 1-weight edge: push the neighbor to the back of the deque.

This ensures that the deque always processes nodes in non-decreasing order of distance, mimicking the behavior of a priority queue without the log factor.

Algorithm Steps

Step 1: Initialize distance array with infinity. Set dist[source] = 0.
Step 2: Push source to deque.
Step 3: While deque is not empty, pop from front.
Step 4: For each neighbor, if new distance is better: update distance. If edge weight = 0, push to front. If edge weight = 1, push to back.

Example: Minimum Flips in Grid

Given a binary grid, find the minimum number of cell flips needed to create a path from top-left to bottom-right. Moving to a cell with value 0 is free; moving to a cell with value 1 costs 1 (flip it).

Implementation

#include <bits/stdc++.h>
using namespace std;

// 0-1 BFS on a graph with edges of weight 0 or 1
vector<int> bfs01(int V, vector<vector<pair<int,int>>>& adj, int src) {
    vector<int> dist(V, INT_MAX);
    dist[src] = 0;
    deque<int> dq;
    dq.push_back(src);

    while (!dq.empty()) {
        int u = dq.front();
        dq.pop_front();

        for (auto& [v, w] : adj[u]) {
            if (dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                if (w == 0)
                    dq.push_front(v);  // 0-weight: front
                else
                    dq.push_back(v);   // 1-weight: back
            }
        }
    }
    return dist;
}

// Example: Minimum flips in binary grid
int minFlips(vector<vector<int>>& grid) {
    int n = grid.size(), m = grid[0].size();
    vector<vector<int>> dist(n, vector<int>(m, INT_MAX));
    dist[0][0] = grid[0][0];
    deque<pair<int,int>> dq;
    dq.push_back({0, 0});

    int dx[] = {0, 0, 1, -1};
    int dy[] = {1, -1, 0, 0};

    while (!dq.empty()) {
        auto [x, y] = dq.front();
        dq.pop_front();

        for (int d = 0; d < 4; d++) {
            int nx = x + dx[d], ny = y + dy[d];
            if (nx < 0 || nx >= n || ny < 0 || ny >= m) continue;

            int w = grid[nx][ny]; // 0 = free, 1 = flip needed
            if (dist[x][y] + w < dist[nx][ny]) {
                dist[nx][ny] = dist[x][y] + w;
                if (w == 0)
                    dq.push_front({nx, ny});
                else
                    dq.push_back({nx, ny});
            }
        }
    }
    return dist[n-1][m-1];
}

int main() {
    vector<vector<int>> grid = {
        {0, 1, 0},
        {0, 1, 0},
        {1, 0, 0}
    };
    cout << minFlips(grid) << endl; // Output: 1
    return 0;
}
import java.util.*;

class ZeroOneBFS {
    // Generic 0-1 BFS on adjacency list
    static int[] bfs01(int V, List<List<int[]>> adj, int src) {
        int[] dist = new int[V];
        Arrays.fill(dist, Integer.MAX_VALUE);
        dist[src] = 0;
        Deque<Integer> dq = new ArrayDeque<>();
        dq.addFirst(src);

        while (!dq.isEmpty()) {
            int u = dq.pollFirst();
            for (int[] edge : adj.get(u)) {
                int v = edge[0], w = edge[1];
                if (dist[u] + w < dist[v]) {
                    dist[v] = dist[u] + w;
                    if (w == 0)
                        dq.addFirst(v);  // 0-weight: front
                    else
                        dq.addLast(v);   // 1-weight: back
                }
            }
        }
        return dist;
    }

    // Minimum flips in binary grid
    static int minFlips(int[][] grid) {
        int n = grid.length, m = grid[0].length;
        int[][] dist = new int[n][m];
        for (int[] row : dist) Arrays.fill(row, Integer.MAX_VALUE);
        dist[0][0] = grid[0][0];

        Deque<int[]> dq = new ArrayDeque<>();
        dq.addFirst(new int[]{0, 0});

        int[] dx = {0, 0, 1, -1};
        int[] dy = {1, -1, 0, 0};

        while (!dq.isEmpty()) {
            int[] curr = dq.pollFirst();
            int x = curr[0], y = curr[1];

            for (int d = 0; d < 4; d++) {
                int nx = x + dx[d], ny = y + dy[d];
                if (nx < 0 || nx >= n || ny < 0 || ny >= m) continue;

                int w = grid[nx][ny];
                if (dist[x][y] + w < dist[nx][ny]) {
                    dist[nx][ny] = dist[x][y] + w;
                    if (w == 0)
                        dq.addFirst(new int[]{nx, ny});
                    else
                        dq.addLast(new int[]{nx, ny});
                }
            }
        }
        return dist[n-1][m-1];
    }

    public static void main(String[] args) {
        int[][] grid = {
            {0, 1, 0},
            {0, 1, 0},
            {1, 0, 0}
        };
        System.out.println(minFlips(grid)); // Output: 1
    }
}
from collections import deque

def bfs_01(V, adj, src):
    """Generic 0-1 BFS. adj[u] = [(v, w), ...] where w is 0 or 1"""
    dist = [float('inf')] * V
    dist[src] = 0
    dq = deque([src])

    while dq:
        u = dq.popleft()
        for v, w in adj[u]:
            if dist[u] + w < dist[v]:
                dist[v] = dist[u] + w
                if w == 0:
                    dq.appendleft(v)  # 0-weight: front
                else:
                    dq.append(v)      # 1-weight: back
    return dist

def min_flips(grid):
    """Minimum flips in binary grid to reach bottom-right"""
    n, m = len(grid), len(grid[0])
    dist = [[float('inf')] * m for _ in range(n)]
    dist[0][0] = grid[0][0]
    dq = deque([(0, 0)])

    directions = [(0,1), (0,-1), (1,0), (-1,0)]

    while dq:
        x, y = dq.popleft()
        for dx, dy in directions:
            nx, ny = x + dx, y + dy
            if 0 <= nx < n and 0 <= ny < m:
                w = grid[nx][ny]  # 0 = free, 1 = flip
                if dist[x][y] + w < dist[nx][ny]:
                    dist[nx][ny] = dist[x][y] + w
                    if w == 0:
                        dq.appendleft((nx, ny))
                    else:
                        dq.append((nx, ny))

    return dist[n-1][m-1]

# Example
grid = [
    [0, 1, 0],
    [0, 1, 0],
    [1, 0, 0]
]
print(min_flips(grid))  # Output: 1

Why It Works

The deque maintains a key invariant: nodes are ordered by non-decreasing distance. Since edge weights are only 0 or 1:

  • A 0-weight edge keeps the same distance level, so the neighbor goes to the front (same priority as current).
  • A 1-weight edge increases distance by 1, so the neighbor goes to the back (next priority level).

This is essentially a two-level BFS where level d and level d+1 coexist in the deque.

Comparison with Dijkstra

Aspect 0-1 BFS Dijkstra
Data structure Deque Min-heap (priority queue)
Time complexity O(V + E) O((V + E) log V)
Edge weights Only 0 and 1 Any non-negative
Space O(V) O(V + E)
When to prefer Binary weight graphs General non-negative weights

Complexity Analysis

  • Time: O(V + E). Each vertex is processed at most once. Each edge is relaxed at most once. No logarithmic factor from heap operations.
  • Space: O(V) for the distance array and deque.
Key Insight: 0-1 BFS can be generalized. If edge weights are 0 and k, you can still use a deque (distances only differ by at most k at any time). For weights 0, 1, 2, ..., k, you need a k-bucket BFS (circular array of k+1 queues).

Common Applications

  • Minimum flips/modifications to create a path in a grid.
  • Minimum walls to break in a maze.
  • Shortest path where some roads are free and others have toll = 1.
  • Minimum operations to convert one state to another (BFS on implicit graph).
