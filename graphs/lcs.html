Longest Common Subsequence

Find the length of the longest subsequence present in both strings. The quintessential 2D DP problem.

Problem

   s1 = "ABCBDAB"
   s2 = "BDCAB"

   Common subsequences: "B", "AB", "BD", "BCB", "BCAB", "BDAB"...
   Longest: "BCAB" (length 4) ✓

   Note: Subsequence ≠ Substring
   Subsequence: characters in order but not necessarily contiguous
   Substring: characters contiguous

State & Transition

State: dp[i][j] = LCS length of s1[0..i-1] and s2[0..j-1]

Transition:
If s1[i-1] == s2[j-1]: dp[i][j] = dp[i-1][j-1] + 1 (characters match!)
Else: dp[i][j] = max(dp[i-1][j], dp[i][j-1]) (skip one character)

Base: dp[0][j] = dp[i][0] = 0 (empty string has LCS 0)

Interactive: Watch Table Fill

Code: LCS

// O(m*n) time, O(m*n) space
int lcs(string& s1, string& s2) {
    int m = s1.size(), n = s2.size();
    vector<vector<int>> dp(m + 1, vector<int>(n + 1, 0));

    for (int i = 1; i <= m; i++) {
        for (int j = 1; j <= n; j++) {
            if (s1[i-1] == s2[j-1]) {
                dp[i][j] = dp[i-1][j-1] + 1;
            } else {
                dp[i][j] = max(dp[i-1][j], dp[i][j-1]);
            }
        }
    }
    return dp[m][n];
}

// Print the actual LCS string
string printLCS(string& s1, string& s2, vector<vector<int>>& dp) {
    string result;
    int i = s1.size(), j = s2.size();
    while (i > 0 && j > 0) {
        if (s1[i-1] == s2[j-1]) {
            result = s1[i-1] + result;
            i--; j--;
        } else if (dp[i-1][j] > dp[i][j-1]) {
            i--;
        } else {
            j--;
        }
    }
    return result;
}

// Space optimized: O(n) space
int lcsOpt(string& s1, string& s2) {
    int m = s1.size(), n = s2.size();
    vector<int> prev(n + 1, 0), curr(n + 1, 0);

    for (int i = 1; i <= m; i++) {
        for (int j = 1; j <= n; j++) {
            if (s1[i-1] == s2[j-1])
                curr[j] = prev[j-1] + 1;
            else
                curr[j] = max(prev[j], curr[j-1]);
        }
        swap(prev, curr);
        fill(curr.begin(), curr.end(), 0);
    }
    return prev[n];
}

// O(m*n) time, O(m*n) space
static int lcs(String s1, String s2) {
    int m = s1.length(), n = s2.length();
    int[][] dp = new int[m + 1][n + 1];

    for (int i = 1; i <= m; i++) {
        for (int j = 1; j <= n; j++) {
            if (s1.charAt(i-1) == s2.charAt(j-1)) {
                dp[i][j] = dp[i-1][j-1] + 1;
            } else {
                dp[i][j] = Math.max(dp[i-1][j], dp[i][j-1]);
            }
        }
    }
    return dp[m][n];
}

// Print LCS
static String printLCS(String s1, String s2, int[][] dp) {
    StringBuilder sb = new StringBuilder();
    int i = s1.length(), j = s2.length();
    while (i > 0 && j > 0) {
        if (s1.charAt(i-1) == s2.charAt(j-1)) {
            sb.insert(0, s1.charAt(i-1));
            i--; j--;
        } else if (dp[i-1][j] > dp[i][j-1]) { i--; }
        else { j--; }
    }
    return sb.toString();
}

# O(m*n) time, O(m*n) space
def lcs(s1, s2):
    m, n = len(s1), len(s2)
    dp = [[0] * (n + 1) for _ in range(m + 1)]

    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if s1[i-1] == s2[j-1]:
                dp[i][j] = dp[i-1][j-1] + 1
            else:
                dp[i][j] = max(dp[i-1][j], dp[i][j-1])
    return dp[m][n]

# Print the LCS
def print_lcs(s1, s2, dp):
    result = []
    i, j = len(s1), len(s2)
    while i > 0 and j > 0:
        if s1[i-1] == s2[j-1]:
            result.append(s1[i-1])
            i -= 1; j -= 1
        elif dp[i-1][j] > dp[i][j-1]:
            i -= 1
        else:
            j -= 1
    return "".join(reversed(result))

# Space optimized: O(n)
def lcs_opt(s1, s2):
    m, n = len(s1), len(s2)
    prev = [0] * (n + 1)
    for i in range(1, m + 1):
        curr = [0] * (n + 1)
        for j in range(1, n + 1):
            if s1[i-1] == s2[j-1]:
                curr[j] = prev[j-1] + 1
            else:
                curr[j] = max(prev[j], curr[j-1])
        prev = curr
    return prev[n]

Related Problems

  • Shortest Common Supersequence: len(s1) + len(s2) - LCS
  • Min Insertions/Deletions: To convert s1→s2: (m - LCS) deletions + (n - LCS) insertions
  • Longest Common Substring: Same but reset to 0 on mismatch (no dp[i-1][j] or dp[i][j-1])
  • Edit Distance: Extension with insert/delete/replace costs

Complexity

  • Time: O(m * n)
  • Space: O(m * n) for full table, O(min(m,n)) with rolling array
  • Path reconstruction: Requires full table (can't space-optimize if printing LCS)
