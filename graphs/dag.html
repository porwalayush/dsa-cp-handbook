DAG Shortest & Longest Path

In a DAG (Directed Acyclic Graph), we can find shortest path in O(V+E) — faster than Dijkstra! The trick: process nodes in topological order.

Why Topological Order Works

  DAG with weights:

  0 --2-→ 1 --3-→ 2 --7-→ 3 --1-→ 4
  |              ↗         ↗
  +-----4------+    +--2--+
  |                 |
  +--------6--------+

  Topological order: 0, 1, 2, 3, 4

  Process in topo order means:
  When we process node X, ALL paths leading TO X have already been considered!
  So dist[X] is already FINAL when we process X.

Algorithm

  1. Do topological sort
  2. Set dist[source] = 0, all others = INF
  3. For each node u in topological order:
    • For each neighbor v of u: relax edge (u→v)
    • Relax: if dist[u] + weight(u,v) < dist[v], update dist[v]

Visual Walkthrough

Graph: 0→1(2), 0→2(4), 0→3(6), 1→2(3), 2→3(2), 3→4(1), 2→4(7)
Source: 0. Topo order: [0, 1, 2, 3, 4]

Init: dist = [0, INF, INF, INF, INF]
Process 0: relax 0→1: dist[1]=2, relax 0→2: dist[2]=4, relax 0→3: dist[3]=6
  dist = [0, 2, 4, 6, INF]
Process 1: relax 1→2: 2+3=5 > 4 (no update)
  dist = [0, 2, 4, 6, INF]
Process 2: relax 2→3: 4+2=6 = 6 (no update), relax 2→4: 4+7=11
  dist = [0, 2, 4, 6, 11]
Process 3: relax 3→4: 6+1=7 < 11 (update!)
  dist = [0, 2, 4, 6, 7] ← FINAL!

Code: Shortest Path in DAG

import java.util.*;

public class DAGShortestPath {
    static void topoSort(int node, List<List<int[]>> adj,
                          boolean[] visited, Stack<Integer> stack) {
        visited[node] = true;
        for (int[] edge : adj.get(node))
            if (!visited[edge[0]])
                topoSort(edge[0], adj, visited, stack);
        stack.push(node);
    }

    static int[] shortestPath(int src, List<List<int[]>> adj, int V) {
        // Step 1: Topological Sort
        boolean[] visited = new boolean[V];
        Stack<Integer> stack = new Stack<>();
        for (int i = 0; i < V; i++)
            if (!visited[i]) topoSort(i, adj, visited, stack);

        // Step 2: Initialize distances
        int[] dist = new int[V];
        Arrays.fill(dist, Integer.MAX_VALUE);
        dist[src] = 0;

        // Step 3: Process in topological order
        while (!stack.isEmpty()) {
            int u = stack.pop();
            if (dist[u] == Integer.MAX_VALUE) continue;

            for (int[] edge : adj.get(u)) {
                int v = edge[0], w = edge[1];
                if (dist[u] + w < dist[v]) {
                    dist[v] = dist[u] + w; // RELAX!
                }
            }
        }
        return dist;
    }

    public static void main(String[] args) {
        int V = 5;
        List<List<int[]>> adj = new ArrayList<>();
        for (int i = 0; i < V; i++) adj.add(new ArrayList<>());

        // adj[u].add({v, weight})
        adj.get(0).add(new int[]{1, 2});
        adj.get(0).add(new int[]{2, 4});
        adj.get(1).add(new int[]{2, 3});
        adj.get(2).add(new int[]{3, 2});
        adj.get(3).add(new int[]{4, 1});

        int[] dist = shortestPath(0, adj, V);
        System.out.println(Arrays.toString(dist)); // [0, 2, 4, 6, 7]
    }
}

void topoSort(int node, vector<vector<pair<int,int>>>& adj,
              vector<bool>& visited, stack<int>& st) {
    visited[node] = true;
    for (auto& [v, w] : adj[node])
        if (!visited[v]) topoSort(v, adj, visited, st);
    st.push(node);
}

vector<int> shortestPath(int src, vector<vector<pair<int,int>>>& adj, int V) {
    // Step 1: Topo sort
    vector<bool> visited(V, false);
    stack<int> st;
    for (int i = 0; i < V; i++)
        if (!visited[i]) topoSort(i, adj, visited, st);

    // Step 2: Init distances
    vector<int> dist(V, INT_MAX);
    dist[src] = 0;

    // Step 3: Process in topo order
    while (!st.empty()) {
        int u = st.top(); st.pop();
        if (dist[u] == INT_MAX) continue;

        for (auto& [v, w] : adj[u]) {
            if (dist[u] + w < dist[v])
                dist[v] = dist[u] + w;
        }
    }
    return dist;
}

from collections import defaultdict

def topo_sort(node, adj, visited, stack):
    visited[node] = True
    for v, w in adj[node]:
        if not visited[v]:
            topo_sort(v, adj, visited, stack)
    stack.append(node)

def shortest_path(src, adj, V):
    # Step 1: Topological sort
    visited = [False] * V
    stack = []
    for i in range(V):
        if not visited[i]:
            topo_sort(i, adj, visited, stack)

    # Step 2: Initialize distances
    dist = [float('inf')] * V
    dist[src] = 0

    # Step 3: Process in topo order
    while stack:
        u = stack.pop()
        if dist[u] == float('inf'):
            continue
        for v, w in adj[u]:
            if dist[u] + w < dist[v]:
                dist[v] = dist[u] + w
    return dist

Longest Path in DAG

Unlike general graphs (NP-hard), longest path in DAG is easy! Just negate all weights and find shortest path, OR use the same approach with max instead of min.

// Longest path: same as shortest but with MAX comparison
static int[] longestPath(int src, List<List<int[]>> adj, int V) {
    boolean[] visited = new boolean[V];
    Stack<Integer> stack = new Stack<>();
    for (int i = 0; i < V; i++)
        if (!visited[i]) topoSort(i, adj, visited, stack);

    int[] dist = new int[V];
    Arrays.fill(dist, Integer.MIN_VALUE);
    dist[src] = 0;

    while (!stack.isEmpty()) {
        int u = stack.pop();
        if (dist[u] == Integer.MIN_VALUE) continue;

        for (int[] edge : adj.get(u)) {
            int v = edge[0], w = edge[1];
            if (dist[u] + w > dist[v]) { // MAX instead of MIN
                dist[v] = dist[u] + w;
            }
        }
    }
    return dist;
}

// Longest path: same logic but maximize
vector<int> longestPath(int src, vector<vector<pair<int,int>>>& adj, int V) {
    vector<bool> visited(V, false);
    stack<int> st;
    for (int i = 0; i < V; i++)
        if (!visited[i]) topoSort(i, adj, visited, st);

    vector<int> dist(V, INT_MIN);
    dist[src] = 0;

    while (!st.empty()) {
        int u = st.top(); st.pop();
        if (dist[u] == INT_MIN) continue;

        for (auto& [v, w] : adj[u]) {
            if (dist[u] + w > dist[v])
                dist[v] = dist[u] + w; // maximize
        }
    }
    return dist;
}

            topo_sort(i, adj, visited, stack)

    dist = [float('-inf')] * V
    dist[src] = 0

        if dist[u] == float('-inf'):

        for v, w in adj[u]:

    return dist

Key Points

  • Time: O(V + E) — much faster than Dijkstra!
  • Works with negative weights (unlike Dijkstra)
  • Only works on DAGs — no cycles allowed
  • Longest path: Just maximize instead of minimize — only possible in DAGs (NP-hard in general)
  • Applications: Critical path method (project scheduling), DP on DAG
