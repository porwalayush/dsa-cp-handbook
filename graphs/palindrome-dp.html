Palindrome DP

Palindrome problems solved with dynamic programming: finding the longest palindromic subsequence, minimum deletions to make a palindrome, and minimum cuts for palindrome partitioning.

1. Longest Palindromic Subsequence (LPS)

Find the length of the longest subsequence that reads the same forwards and backwards.

   s = "bbbab"

   Palindromic subsequences: "b", "bb", "bbb", "bbbb", "bab", "bbb"
   Longest: "bbbb" (length 4)

   Key insight: LPS(s) = LCS(s, reverse(s))
   OR solve directly with interval DP on substrings

State & Transition

State: dp[i][j] = length of LPS in s[i..j]

Transition:
If s[i] == s[j]: dp[i][j] = dp[i+1][j-1] + 2
Else: dp[i][j] = max(dp[i+1][j], dp[i][j-1])

Base Cases:
dp[i][i] = 1 (single character is a palindrome)
dp[i][j] = 0 if i > j (empty string)

Visual Walkthrough

   s = "bbbab"   (indices 0..4)

   Fill diagonally (increasing substring length):

   Length 1: dp[i][i] = 1 for all i
   dp = [[1,_,_,_,_],
         [_,1,_,_,_],
         [_,_,1,_,_],
         [_,_,_,1,_],
         [_,_,_,_,1]]

   Length 2: s[0]=s[1]='b' → dp[0][1] = dp[1][0]+2 = 2
             s[1]=s[2]='b' → dp[1][2] = 2
             s[2]='b', s[3]='a' → dp[2][3] = max(dp[3][3],dp[2][2]) = 1
             s[3]='a', s[4]='b' → dp[3][4] = 1

   Length 3: s[0]='b',s[2]='b' → dp[0][2] = dp[1][1]+2 = 3
             s[1]='b',s[3]='a' → dp[1][3] = max(dp[2][3],dp[1][2]) = 2
             s[2]='b',s[4]='b' → dp[2][4] = dp[3][3]+2 = 3

   Length 4: s[0]='b',s[3]='a' → dp[0][3] = max(dp[1][3],dp[0][2]) = 3
             s[1]='b',s[4]='b' → dp[1][4] = dp[2][3]+2 = 3

   Length 5: s[0]='b',s[4]='b' → dp[0][4] = dp[1][3]+2 = 4

   Answer: dp[0][4] = 4 (subsequence "bbbb")

2. Minimum Deletions to Make Palindrome

Key insight: Min deletions = n - LPS(s)

The longest palindromic subsequence is what we KEEP.
Everything else must be deleted.

Example: s = "aebcbda", n = 7, LPS = "abcba" (length 5)
Min deletions = 7 - 5 = 2 (delete 'e' and 'd')

3. Palindrome Partitioning (Min Cuts)

Given a string, find the minimum number of cuts needed so that every partition is a palindrome.

   s = "aab"
   Partitions:  "a"|"a"|"b" = 2 cuts
                "aa"|"b"    = 1 cut   ← optimal

   s = "abcba"
   Already a palindrome → 0 cuts

   s = "abcde"
   "a"|"b"|"c"|"d"|"e" = 4 cuts (worst case, all single chars)

State & Transition

Step 1: Precompute isPalin[i][j] = is s[i..j] a palindrome?

Step 2: dp[i] = min cuts for s[0..i]

Transition:
If s[0..i] is a palindrome: dp[i] = 0
Else: dp[i] = min(dp[j-1] + 1) for all j where s[j..i] is a palindrome

Base Case: dp[0] = 0 (single char, no cut needed) 
   s = "aab"  (indices 0, 1, 2)

   isPalin table:
         0    1    2
   0  [  T    T    F  ]     "a"=T, "aa"=T, "aab"=F
   1  [  -    T    F  ]     "a"=T, "ab"=F
   2  [  -    -    T  ]     "b"=T

   dp[0] = 0   ("a" is palindrome)
   dp[1] = 0   ("aa" is palindrome, isPalin[0][1]=T)
   dp[2]: check all j where isPalin[j][2] = T
          j=2: isPalin[2][2]=T → dp[1]+1 = 1
          j=0: isPalin[0][2]=F → skip
          dp[2] = 1

   Answer: 1 cut ("aa"|"b")

Code: Longest Palindromic Subsequence

// LPS using interval DP: O(n^2) time, O(n^2) space
int longestPalinSubseq(string s) {
    int n = s.size();
    vector<vector<int>> dp(n, vector<int>(n, 0));

    // Base: single characters
    for (int i = 0; i < n; i++) dp[i][i] = 1;

    // Fill by increasing length
    for (int len = 2; len <= n; len++) {
        for (int i = 0; i <= n - len; i++) {
            int j = i + len - 1;
            if (s[i] == s[j]) {
                dp[i][j] = dp[i+1][j-1] + 2;
            } else {
                dp[i][j] = max(dp[i+1][j], dp[i][j-1]);
            }
        }
    }
    return dp[0][n-1];
}

// Space optimized to O(n) using two rows
int lpsOptimized(string s) {
    int n = s.size();
    vector<int> dp(n, 1), prev(n, 0);

    for (int i = n - 2; i >= 0; i--) {
        prev = dp;
        for (int j = i + 1; j < n; j++) {
            if (s[i] == s[j]) {
                dp[j] = prev[j-1] + 2;
            } else {
                dp[j] = max(dp[j], dp[j-1]);
            }
        }
    }
    return dp[n-1];
}

def longest_palin_subseq(s):
    n = len(s)
    dp = [[0] * n for _ in range(n)]

    for i in range(n):
        dp[i][i] = 1

    for length in range(2, n + 1):
        for i in range(n - length + 1):
            j = i + length - 1
            if s[i] == s[j]:
                dp[i][j] = dp[i+1][j-1] + 2
            else:
                dp[i][j] = max(dp[i+1][j], dp[i][j-1])
    return dp[0][n-1]

def lps_via_lcs(s):

Code: Palindrome Partitioning (Min Cuts)

// Min cuts for palindrome partitioning: O(n^2)
int minCut(string s) {
    int n = s.size();

    // Step 1: Precompute palindrome table
    vector<vector<bool>> isPalin(n, vector<bool>(n, false));
    for (int i = n - 1; i >= 0; i--) {
        for (int j = i; j < n; j++) {
            if (s[i] == s[j] && (j - i <= 2 || isPalin[i+1][j-1]))
                isPalin[i][j] = true;
        }
    }

    // Step 2: dp[i] = min cuts for s[0..i]
    vector<int> dp(n, 0);
    for (int i = 1; i < n; i++) {
        if (isPalin[0][i]) {
            dp[i] = 0; // whole prefix is palindrome
            continue;
        }
        dp[i] = i; // worst case: cut every char
        for (int j = 1; j <= i; j++) {
            if (isPalin[j][i]) {
                dp[i] = min(dp[i], dp[j-1] + 1);
            }
        }
    }
    return dp[n-1];
}

// Min cuts for palindrome partitioning: O(n^2)
static int minCut(String s) {
    int n = s.length();

    // Step 1: Precompute palindrome table
    boolean[][] isPalin = new boolean[n][n];
    for (int i = n - 1; i >= 0; i--) {
        for (int j = i; j < n; j++) {
            if (s.charAt(i) == s.charAt(j) &&
                (j - i <= 2 || isPalin[i+1][j-1]))
                isPalin[i][j] = true;
        }
    }

    // Step 2: dp[i] = min cuts for s[0..i]
    int[] dp = new int[n];
    for (int i = 1; i < n; i++) {
        if (isPalin[0][i]) { dp[i] = 0; continue; }
        dp[i] = i;
        for (int j = 1; j <= i; j++) {
            if (isPalin[j][i]) {
                dp[i] = Math.min(dp[i], dp[j-1] + 1);
            }
        }
    }
    return dp[n-1];
}

# Min cuts for palindrome partitioning: O(n^2)
def min_cut(s):
    n = len(s)

    # Step 1: Precompute palindrome table
    is_palin = [[False] * n for _ in range(n)]
    for i in range(n - 1, -1, -1):
        for j in range(i, n):
            if s[i] == s[j] and (j - i <= 2 or is_palin[i+1][j-1]):
                is_palin[i][j] = True

    # Step 2: dp[i] = min cuts for s[0..i]
    dp = list(range(n))  # worst case: i cuts for s[0..i]
    for i in range(1, n):
        if is_palin[0][i]:
            dp[i] = 0
            continue
        for j in range(1, i + 1):
            if is_palin[j][i]:
                dp[i] = min(dp[i], dp[j-1] + 1)
    return dp[n-1]

   ┌──────────────────────────────────────────────────────┐
   │        Palindrome DP Problem Hierarchy                │
   ├──────────────────────────────────────────────────────┤
   │                                                       │
   │  LPS (Longest Palindromic Subsequence)                │
   │   ↓                                                   │
   │  Min Deletions = n - LPS                              │
   │   ↓                                                   │
   │  Min Insertions to make palindrome = n - LPS          │
   │   (same as min deletions!)                            │
   │                                                       │
   │  Palindrome Partitioning                              │
   │   = precompute isPalin[][] + 1D DP for min cuts       │
   │                                                       │
   │  Longest Palindromic Substring (different!)           │
   │   = expand from center, O(n^2) or Manacher O(n)       │
   └──────────────────────────────────────────────────────┘

Complexity Summary

ProblemTimeSpace
Longest Palindromic SubsequenceO(n^2)O(n^2) or O(n)

Palindrome Partitioning (min cuts)O(n^2)O(n^2)
Longest Palindromic SubstringO(n^2) or O(n)O(n^2) or O(1)
  • LPS uses interval DP (fill by increasing substring length) or LCS with reversed string
  • Palindrome partitioning requires two phases: precompute + linear scan
  • Space for LPS can be reduced to O(n) by processing rows bottom-up and keeping only current + previous row
  • Don't confuse subsequence (chars can be non-contiguous) with substring (must be contiguous)
