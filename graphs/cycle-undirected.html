Cycle Detection in Undirected Graph

A cycle exists if we can start from a node and come back to it without retracing our steps.

The Idea

  Has Cycle:                No Cycle (Tree):

    0 --- 1                    0
    |     |                   / \
    |     |                  1   2
    3 --- 2                 / \
                           3   4
  0→1→2→3→0 = CYCLE!

Method 1: DFS with Parent Tracking

Key Insight: During DFS, if we find a visited node that is NOT our parent, we found a cycle!

  DFS from 0:
  Visit 0 (parent = -1)
  Visit 1 (parent = 0)
  Visit 2 (parent = 1)
  Visit 3 (parent = 2) → neighbor 0 is visited AND not parent(2) → CYCLE!

  Why check parent? Because in undirected graph:
  0 --- 1 means both 0→1 and 1→0 exist
  When we go 0→1, neighbor 0 is visited but it's the parent, so it's NOT a cycle

import java.util.*;

public class CycleUndirected {
    static boolean hasCycleDFS(int node, int parent,
                               List<List<Integer>> adj, boolean[] visited) {
        visited[node] = true;

        for (int neighbor : adj.get(node)) {
            if (!visited[neighbor]) {
                // Go deeper
                if (hasCycleDFS(neighbor, node, adj, visited))
                    return true;
            }
            else if (neighbor != parent) {
                // Visited node that's not our parent = CYCLE!
                return true;
            }
        }
        return false;
    }

    static boolean hasCycle(List<List<Integer>> adj, int V) {
        boolean[] visited = new boolean[V];

        // Check each component (graph might be disconnected)
        for (int i = 0; i < V; i++) {
            if (!visited[i]) {
                if (hasCycleDFS(i, -1, adj, visited))
                    return true;
            }
        }
        return false;
    }

    public static void main(String[] args) {
        int V = 4;
        List<List<Integer>> adj = new ArrayList<>();
        for (int i = 0; i < V; i++) adj.add(new ArrayList<>());

        // 0-1-2-3-0 (has cycle)
        adj.get(0).add(1); adj.get(1).add(0);
        adj.get(1).add(2); adj.get(2).add(1);
        adj.get(2).add(3); adj.get(3).add(2);
        adj.get(3).add(0); adj.get(0).add(3);

        System.out.println("Has cycle: " + hasCycle(adj, V)); // true
    }
}

def has_cycle_dfs(node, parent, adj, visited):
    visited[node] = True

    for neighbor in adj[node]:
        if not visited[neighbor]:
            if has_cycle_dfs(neighbor, node, adj, visited):
                return True
        elif neighbor != parent:
            return True  # CYCLE!
    return False

def has_cycle(adj, V):
    visited = [False] * V
    for i in range(V):
        if not visited[i]:
            if has_cycle_dfs(i, -1, adj, visited):
                return True
    return False

Method 2: Using DSU (Union-Find)

Key Insight: Process edges one by one. If both endpoints are already in the same set → adding this edge creates a cycle!

  Edges: (0,1), (1,2), (2,3), (3,0)

  Process (0,1): union(0,1) → Set: {0,1}
  Process (1,2): union(1,2) → Set: {0,1,2}
  Process (2,3): union(2,3) → Set: {0,1,2,3}
  Process (3,0): find(3)=0, find(0)=0 → SAME SET → CYCLE!

public class CycleUnionFind {
    static int[] parent, rank;

    static int find(int x) {
        if (parent[x] != x)
            parent[x] = find(parent[x]); // path compression
        return parent[x];
    }

    static boolean union(int x, int y) {
        int px = find(x), py = find(y);
        if (px == py) return false; // already same set = cycle!

        if (rank[px] < rank[py]) parent[px] = py;
        else if (rank[px] > rank[py]) parent[py] = px;
        else { parent[py] = px; rank[px]++; }
        return true;
    }

    static boolean hasCycle(int[][] edges, int V) {
        parent = new int[V];
        rank = new int[V];
        for (int i = 0; i < V; i++) parent[i] = i;

        for (int[] edge : edges) {
            if (!union(edge[0], edge[1]))
                return true; // cycle detected!
        }
        return false;
    }
}

#include <bits/stdc++.h>
using namespace std;

vector<int> parent, rnk;

int find(int x) {
    if (parent[x] != x)
        parent[x] = find(parent[x]);
    return parent[x];
}

bool unite(int x, int y) {
    int px = find(x), py = find(y);
    if (px == py) return false; // cycle!

    if (rnk[px] < rnk[py]) parent[px] = py;
    else if (rnk[px] > rnk[py]) parent[py] = px;
    else { parent[py] = px; rnk[px]++; }
    return true;
}

bool hasCycle(vector<pair<int,int>>& edges, int V) {
    parent.resize(V); rnk.resize(V, 0);
    iota(parent.begin(), parent.end(), 0);

    for (auto& [u, v] : edges) {
        if (!unite(u, v)) return true;
    }
    return false;
}

parent = []
rank = []

def find(x):
    if parent[x] != x:
        parent[x] = find(parent[x])  # path compression
    return parent[x]

def union(x, y):
    px, py = find(x), find(y)
    if px == py:
        return False  # cycle!
    if rank[px] < rank[py]:
        parent[px] = py
    elif rank[px] > rank[py]:
        parent[py] = px
    else:
        parent[py] = px
        rank[px] += 1
    return True

def has_cycle(edges, V):
    global parent, rank
    parent = list(range(V))
    rank = [0] * V
    for u, v in edges:
        if not union(u, v):
            return True
    return False

Quick Summary

  • DFS + Parent: Easy to implement, works component-wise. O(V+E)
  • Union-Find: Process edge-by-edge. Great when edges come one at a time. Nearly O(E)
  • Simple rule: Undirected graph with V vertices → if edges >= V, guaranteed to have a cycle
