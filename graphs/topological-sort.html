Topological Sort (DFS Method)

A linear ordering of vertices such that for every directed edge u→v, u comes before v. Only works on DAGs (Directed Acyclic Graphs).

Real-Life Example

  Course Prerequisites:

  Math101 → Math201 → Math301
       ↘               ↗
        Physics101 → Physics201

  Valid order: Math101, Physics101, Math201, Physics201, Math301

  Rule: You must complete prerequisites BEFORE taking the course.
  Topological sort gives you a valid order to take all courses!

DFS Approach

Key Insight: After DFS finishes a node (all its descendants are processed), push it to a stack. The stack gives topological order when popped!

  Graph: 5→0, 5→2, 4→0, 4→1, 2→3, 3→1

    5 → 0      4 → 1
    ↓           ↑
    2 → 3 ------+

  DFS finish order (when node turns BLACK): 0, 1, 3, 2, 4, 5
  Stack (push on finish): bottom [0, 1, 3, 2, 4, 5] top

  Pop stack → Topological order: 5, 4, 2, 3, 1, 0 ✓
  (Every edge goes left-to-right in this ordering)

import java.util.*;

public class TopoSort {
    static void dfs(int node, List<List<Integer>> adj,
                    boolean[] visited, Stack<Integer> stack) {
        visited[node] = true;

        // First, visit all dependencies (neighbors)
        for (int neighbor : adj.get(node)) {
            if (!visited[neighbor]) {
                dfs(neighbor, adj, visited, stack);
            }
        }

        // AFTER all neighbors are done, push this node
        stack.push(node);
    }

    static List<Integer> topologicalSort(List<List<Integer>> adj, int V) {
        boolean[] visited = new boolean[V];
        Stack<Integer> stack = new Stack<>();

        for (int i = 0; i < V; i++) {
            if (!visited[i]) {
                dfs(i, adj, visited, stack);
            }
        }

        // Pop stack to get topological order
        List<Integer> result = new ArrayList<>();
        while (!stack.isEmpty()) {
            result.add(stack.pop());
        }
        return result;
    }

    public static void main(String[] args) {
        int V = 6;
        List<List<Integer>> adj = new ArrayList<>();
        for (int i = 0; i < V; i++) adj.add(new ArrayList<>());

        adj.get(5).add(0); adj.get(5).add(2);
        adj.get(4).add(0); adj.get(4).add(1);
        adj.get(2).add(3); adj.get(3).add(1);

        System.out.println("Topological Sort: " + topologicalSort(adj, V));
    }
}

#include <bits/stdc++.h>
using namespace std;

void dfs(int node, vector<vector<int>>& adj,
         vector<bool>& visited, stack<int>& st) {
    visited[node] = true;

    for (int neighbor : adj[node]) {
        if (!visited[neighbor]) {
            dfs(neighbor, adj, visited, st);
        }
    }
    st.push(node); // push AFTER all neighbors done
}

vector<int> topologicalSort(vector<vector<int>>& adj, int V) {
    vector<bool> visited(V, false);
    stack<int> st;

    for (int i = 0; i < V; i++) {
        if (!visited[i]) dfs(i, adj, visited, st);
    }

    vector<int> result;
    while (!st.empty()) {
        result.push_back(st.top());
        st.pop();
    }
    return result;
}

int main() {
    int V = 6;
    vector<vector<int>> adj(V);
    adj[5].push_back(0); adj[5].push_back(2);
    adj[4].push_back(0); adj[4].push_back(1);
    adj[2].push_back(3); adj[3].push_back(1);

    vector<int> result = topologicalSort(adj, V);
    cout << "Topological Sort: ";
    for (int x : result) cout << x << " ";
    return 0;
}

def dfs(node, adj, visited, stack):
    visited[node] = True
    for neighbor in adj[node]:
        if not visited[neighbor]:
            dfs(neighbor, adj, visited, stack)
    stack.append(node)  # push AFTER all neighbors done

def topological_sort(adj, V):
    visited = [False] * V
    stack = []

    for i in range(V):
        if not visited[i]:
            dfs(i, adj, visited, stack)

    return stack[::-1]  # reverse = topo order

# Example
V = 6
adj = [[] for _ in range(V)]
adj[5] = [0, 2]; adj[4] = [0, 1]
adj[2] = [3];    adj[3] = [1]

print("Topological Sort:", topological_sort(adj, V))

Key Points

  • Only works on DAGs — if graph has a cycle, topological sort is impossible
  • Not unique — multiple valid orderings can exist
  • Why push AFTER? A node should come before its dependencies. After DFS finishes all children, that node has no more "things to do first" → safe to add it.
  • Time: O(V + E)
  • Applications: Build systems, task scheduling, course prerequisites
