Implicit Graphs

An implicit graph is one where you don't build an adjacency list. Instead, neighbors are computed on-the-fly using rules. Most grid problems and state-space problems are implicit graphs!

What Makes a Graph "Implicit"?

  EXPLICIT GRAPH:                    IMPLICIT GRAPH:
  You build adjacency list:          No adj list needed!
  adj[0] = [1, 2]                    Grid cell (r,c) → neighbors are
  adj[1] = [0, 3]                    (r+1,c), (r-1,c), (r,c+1), (r,c-1)
  adj[2] = [0]
                                     Rule defines the edges!

Type 1: Grid as Graph (Most Common)

Each cell is a node. Edges go to 4 (or 8) adjacent cells.

  Grid:                    As a Graph:
  S . . .                  S ─ . ─ . ─ .
  . # # .                  |   |   |   |
  . . . E                  . ─ # ─ # ─ .
  . # . .                  |   |   |   |
                           . ─ . ─ . ─ E
  S = start, E = end       |   |   |   |
  # = wall (no edge)       . ─ # ─ . ─ .

  BFS on this grid = shortest path from S to E!

Grid BFS (Shortest Path in Maze)

import java.util.*;

public class GridBFS {
    static int shortestPath(char[][] grid) {
        int R = grid.length, C = grid[0].length;
        int sr = -1, sc = -1;

        // Find start position
        for (int i = 0; i < R; i++)
            for (int j = 0; j < C; j++)
                if (grid[i][j] == 'S') { sr = i; sc = j; }

        // BFS
        int[][] dist = new int[R][C];
        for (int[] row : dist) Arrays.fill(row, -1);
        dist[sr][sc] = 0;

        Queue<int[]> queue = new LinkedList<>();
        queue.add(new int[]{sr, sc});

        // 4 directions: up, down, left, right
        int[][] dirs = {{-1,0},{1,0},{0,-1},{0,1}};

        while (!queue.isEmpty()) {
            int[] cell = queue.poll();
            int r = cell[0], c = cell[1];

            for (int[] d : dirs) {
                int nr = r + d[0], nc = c + d[1];

                // Check bounds, not wall, not visited
                if (nr >= 0 && nr < R && nc >= 0 && nc < C
                    && grid[nr][nc] != '#' && dist[nr][nc] == -1) {

                    dist[nr][nc] = dist[r][c] + 1;

                    if (grid[nr][nc] == 'E')
                        return dist[nr][nc]; // found end!

                    queue.add(new int[]{nr, nc});
                }
            }
        }
        return -1; // no path
    }
}

#include <bits/stdc++.h>
using namespace std;

int shortestPath(vector<vector<char>>& grid) {
    int R = grid.size(), C = grid[0].size();
    int sr = -1, sc = -1;

    for (int i = 0; i < R; i++)
        for (int j = 0; j < C; j++)
            if (grid[i][j] == 'S') { sr = i; sc = j; }

    vector<vector<int>> dist(R, vector<int>(C, -1));
    dist[sr][sc] = 0;

    queue<pair<int,int>> q;
    q.push({sr, sc});

    int dx[] = {-1,1,0,0}, dy[] = {0,0,-1,1};

    while (!q.empty()) {
        auto [r, c] = q.front(); q.pop();

        for (int d = 0; d < 4; d++) {
            int nr = r + dx[d], nc = c + dy[d];

            if (nr >= 0 && nr < R && nc >= 0 && nc < C
                && grid[nr][nc] != '#' && dist[nr][nc] == -1) {

                dist[nr][nc] = dist[r][c] + 1;
                if (grid[nr][nc] == 'E') return dist[nr][nc];
                q.push({nr, nc});
            }
        }
    }
    return -1;
}

from collections import deque

def shortest_path(grid):
    R, C = len(grid), len(grid[0])
    sr, sc = -1, -1

    # Find start
    for i in range(R):
        for j in range(C):
            if grid[i][j] == 'S':
                sr, sc = i, j

    dist = [[-1] * C for _ in range(R)]
    dist[sr][sc] = 0
    queue = deque([(sr, sc)])
    dirs = [(-1,0), (1,0), (0,-1), (0,1)]

    while queue:
        r, c = queue.popleft()
        for dr, dc in dirs:
            nr, nc = r + dr, c + dc
            if 0 <= nr < R and 0 <= nc < C and grid[nr][nc] != '#' and dist[nr][nc] == -1:
                dist[nr][nc] = dist[r][c] + 1
                if grid[nr][nc] == 'E':
                    return dist[nr][nc]
                queue.append((nr, nc))
    return -1

Type 2: State-Space Graph

Each "state" is a node. Transitions between states are edges. Examples: word ladder, combination lock, puzzle states.

  Word Ladder: "hit" → "hot" → "dot" → "dog" → "cog"

  Each word is a node.
  Edge exists if words differ by exactly 1 letter.

  "hit" → "hot" (change i→o)
  "hot" → "dot" (change h→d)
  "dot" → "dog" (change t→g)
  "dog" → "cog" (change d→c)

  BFS gives shortest transformation sequence!

Type 3: Knight Moves on Chessboard

  Knight at (0,0), find minimum moves to reach (7,7)

  From any position (r,c), knight can move to:
  (r±2, c±1) or (r±1, c±2)   ← 8 possible moves

  This is an implicit graph:
  - Nodes: all board positions
  - Edges: valid knight moves
  - BFS gives minimum moves!

// Minimum knight moves from (0,0) to (tr,tc) on N×N board
static int minKnightMoves(int N, int tr, int tc) {
    int[][] moves = {{-2,-1},{-2,1},{-1,-2},{-1,2},
                     {1,-2},{1,2},{2,-1},{2,1}};

    boolean[][] visited = new boolean[N][N];
    Queue<int[]> queue = new LinkedList<>();
    queue.add(new int[]{0, 0, 0}); // {row, col, moves}
    visited[0][0] = true;

    while (!queue.isEmpty()) {
        int[] curr = queue.poll();
        if (curr[0] == tr && curr[1] == tc)
            return curr[2];

        for (int[] m : moves) {
            int nr = curr[0] + m[0], nc = curr[1] + m[1];
            if (nr >= 0 && nr < N && nc >= 0 && nc < N && !visited[nr][nc]) {
                visited[nr][nc] = true;
                queue.add(new int[]{nr, nc, curr[2] + 1});
            }
        }
    }
    return -1;
}

// Minimum knight moves from (0,0) to (tr,tc) on N×N board
int minKnightMoves(int N, int tr, int tc) {
    int dx[] = {-2,-2,-1,-1,1,1,2,2};
    int dy[] = {-1,1,-2,2,-2,2,-1,1};

    vector<vector<bool>> visited(N, vector<bool>(N, false));
    queue<tuple<int,int,int>> q;
    q.push({0, 0, 0});
    visited[0][0] = true;

    while (!q.empty()) {
        auto [r, c, moves] = q.front(); q.pop();
        if (r == tr && c == tc) return moves;

        for (int d = 0; d < 8; d++) {
            int nr = r + dx[d], nc = c + dy[d];
            if (nr >= 0 && nr < N && nc >= 0 && nc < N && !visited[nr][nc]) {
                visited[nr][nc] = true;
                q.push({nr, nc, moves + 1});
            }
        }
    }
    return -1;
}

from collections import deque

def min_knight_moves(N, tr, tc):
    moves = [(-2,-1),(-2,1),(-1,-2),(-1,2),(1,-2),(1,2),(2,-1),(2,1)]
    visited = [[False]*N for _ in range(N)]
    queue = deque([(0, 0, 0)])  # (row, col, num_moves)
    visited[0][0] = True

    while queue:
        r, c, steps = queue.popleft()
        if r == tr and c == tc:
            return steps
        for dr, dc in moves:
            nr, nc = r + dr, c + dc
            if 0 <= nr < N and 0 <= nc < N and not visited[nr][nc]:
                visited[nr][nc] = True
                queue.append((nr, nc, steps + 1))
    return -1

Key Patterns

  • Grid → Graph: 4 directions = 4 edges per cell. Wall = no edge.
  • State → Graph: Each unique state is a node. Valid transitions = edges.
  • Visited tracking: Use 2D array for grids, HashSet for states
  • Common problems: Maze shortest path, word ladder, sliding puzzle, minimum operations to reach target
  • Advantage: No need to explicitly build graph — saves memory!
