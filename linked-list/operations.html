Insert, Delete & Traverse

Every linked list operation boils down to one skill: rewiring next pointers in the right order. If you do it in the wrong order you cut the chain and lose nodes. Let's make that order visible.

Traversal — walking the chain

You can't jump to the i-th node like an array. You start at head and follow next until you hit null. This O(n) walk is the foundation of almost every other operation.

void traverse(Node* head) {
    Node* curr = head;
    while (curr != nullptr) {
        cout << curr->data << " -> ";
        curr = curr->next;        // step forward
    }
    cout << "null";
}

void traverse(Node head) {
    Node curr = head;
    while (curr != null) {
        System.out.print(curr.data + " -> ");
        curr = curr.next;         // step forward
    }
    System.out.println("null");
}

def traverse(head):
    curr = head
    while curr:
        print(curr.data, end=" -> ")
        curr = curr.next          # step forward
    print("None")

Insertion

To insert a node, you create it, point its next at what should follow it, and only then point the previous node at the new one. Order matters — link the newcomer forward before you redirect the predecessor.

Inserting at the head is the easy case and O(1):
  • newNode.next = head
  • head = newNode
For the middle, walk to the node before the target position first. 
// Insert value at index (0 = head). Returns new head.
Node* insertAt(Node* head, int index, int value) {
    Node* node = new Node(value);
    if (index == 0) {        // insert at head
        node->next = head;
        return node;
    }
    Node* prev = head;
    for (int i = 0; i < index - 1 && prev; i++)
        prev = prev->next;        // stop at node before target
    node->next = prev->next;       // 1) link newcomer forward
    prev->next = node;            // 2) redirect predecessor
    return head;
}

// Insert value at index (0 = head). Returns new head.
Node insertAt(Node head, int index, int value) {
    Node node = new Node(value);
    if (index == 0) {
        node.next = head;
        return node;
    }
    Node prev = head;
    for (int i = 0; i < index - 1 && prev != null; i++)
        prev = prev.next;
    node.next = prev.next;        // 1) link forward
    prev.next = node;             // 2) redirect
    return head;
}

# Insert value at index (0 = head). Returns new head.
def insert_at(head, index, value):
    node = Node(value)
    if index == 0:
        node.next = head
        return node
    prev = head
    for _ in range(index - 1):
        if prev: prev = prev.next
    node.next = prev.next          # 1) link forward
    prev.next = node               # 2) redirect
    return head

Deletion

To delete, you make the previous node "skip over" the target: point prev.next at target.next. The target is now unreachable (in C++, free it to avoid a leak).

// Delete first node with given value. Returns new head.
Node* deleteVal(Node* head, int value) {
    if (!head) return nullptr;
    if (head->data == value) {     // deleting the head
        Node* nxt = head->next;
        delete head;
        return nxt;
    }
    Node* prev = head;
    while (prev->next && prev->next->data != value)
        prev = prev->next;
    if (prev->next) {
        Node* target = prev->next;
        prev->next = target->next;  // skip over target
        delete target;
    }
    return head;
}

// Delete first node with given value. Returns new head.
Node deleteVal(Node head, int value) {
    if (head == null) return null;
    if (head.data == value) return head.next;
    Node prev = head;
    while (prev.next != null && prev.next.data != value)
        prev = prev.next;
    if (prev.next != null)
        prev.next = prev.next.next;  // skip over target
    return head;
}

# Delete first node with given value. Returns new head.
def delete_val(head, value):
    if not head: return None
    if head.data == value:
        return head.next
    prev = head
    while prev.next and prev.next.data != value:
        prev = prev.next
    if prev.next:
        prev.next = prev.next.next   # skip over target
    return head
The dummy-node trick: deleting/inserting at the head is always a special case. To kill that special case, create a fake node dummy.next = head, do all your work from dummy, and return dummy.next at the end. You'll see this everywhere in the harder problems.
