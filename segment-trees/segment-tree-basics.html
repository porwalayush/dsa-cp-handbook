Segment Tree: Build & Query

A Segment Tree is a binary tree where each node stores aggregate information (sum, min, max, GCD) for a range of the array. It supports both point updates and range queries in O(log n).

Structure & Indexing

Array: [2, 1, 5, 3, 4, 7, 6, 8]  (n = 8)

Segment Tree (sum):
                     [36]                  node 1: [0,7]
                   /      \
              [11]           [25]          nodes 2,3
             /    \         /    \
          [3]     [8]    [11]    [14]     nodes 4-7
          / \     / \     / \     / \
        [2] [1] [5] [3] [4] [7] [6] [8]  nodes 8-15 (leaves)

Node i:
  Left child  = 2*i
  Right child = 2*i + 1
  Parent      = i / 2

Node i represents range [l, r]:
  Left child covers [l, mid]
  Right child covers [mid+1, r]
  where mid = (l + r) / 2

Array size needed: 4 * n (safe upper bound)

Build Operation

Recursively build bottom-up. Leaves hold array values; internal nodes combine their children.

build(node, l, r):
  if l == r:
    tree[node] = arr[l]    // leaf
    return
  mid = (l + r) / 2
  build(2*node, l, mid)    // build left
  build(2*node+1, mid+1, r) // build right
  tree[node] = tree[2*node] + tree[2*node+1]  // combine

Time: O(n)  |  Space: O(4n)

Point Update

Update arr[2] from 5 to 9 (add 4):

                     [36+4=40]
                   /      \
              [11+4=15]    [25]
             /    \
          [3]     [8+4=12]
                  / \
                [5+4=9] [3]

Path: root -> left -> right -> leaf at index 2
Only O(log n) nodes are modified (ancestors of the leaf)

Range Query

Query: sum(1, 5) = sum of arr[1..5] (0-indexed)

                     [36]  <- covers [0,7], partial overlap
                   /      \
              [11]          [25]  <- covers [4,7], partial overlap
             /    \         /
          [3]     [8]    [11]    <- covers [4,5], fully inside
          / \     / \
        [2] [1] [5] [3]

Traversal:
  Node 1 [0,7]: partial overlap with [1,5] -> recurse both
    Node 2 [0,3]: partial overlap -> recurse both
      Node 4 [0,1]: partial overlap -> recurse both
        Node 8 [0,0]: outside [1,5] -> return 0
        Node 9 [1,1]: fully inside -> return 1
      Node 5 [2,3]: fully inside [1,5] -> return 8
    Node 3 [4,7]: partial overlap -> recurse both
      Node 6 [4,5]: fully inside [1,5] -> return 11
      Node 7 [6,7]: outside [1,5] -> return 0

Answer = 0 + 1 + 8 + 11 + 0 = 20 = 1+5+3+4+7 ✓

Code Implementation

#include <bits/stdc++.h>
using namespace std;

struct SegTree {
    vector<int> tree;
    int n;

    SegTree(vector<int>& arr) {
        n = arr.size();
        tree.resize(4 * n);
        build(arr, 1, 0, n - 1);
    }

    void build(vector<int>& arr, int node, int l, int r) {
        if (l == r) {
            tree[node] = arr[l];
            return;
        }
        int mid = (l + r) / 2;
        build(arr, 2 * node, l, mid);
        build(arr, 2 * node + 1, mid + 1, r);
        tree[node] = tree[2 * node] + tree[2 * node + 1];
    }

    // Point update: set arr[idx] = val
    void update(int node, int l, int r, int idx, int val) {
        if (l == r) {
            tree[node] = val;
            return;
        }
        int mid = (l + r) / 2;
        if (idx <= mid)
            update(2 * node, l, mid, idx, val);
        else
            update(2 * node + 1, mid + 1, r, idx, val);
        tree[node] = tree[2 * node] + tree[2 * node + 1];
    }

    void update(int idx, int val) { update(1, 0, n - 1, idx, val); }

    // Range query: sum of [ql, qr]
    int query(int node, int l, int r, int ql, int qr) {
        if (qr < l || r < ql) return 0;       // no overlap
        if (ql <= l && r <= qr) return tree[node]; // full overlap
        int mid = (l + r) / 2;
        return query(2 * node, l, mid, ql, qr) +
               query(2 * node + 1, mid + 1, r, ql, qr);
    }

    int query(int ql, int qr) { return query(1, 0, n - 1, ql, qr); }
};

int main() {
    vector<int> arr = {2, 1, 5, 3, 4, 7, 6, 8};
    SegTree st(arr);
    cout << st.query(1, 5) << endl;  // 20
    st.update(2, 9);                  // arr[2] = 9
    cout << st.query(1, 5) << endl;  // 24
}

class SegTree {
    int[] tree;
    int n;

    SegTree(int[] arr) {
        n = arr.length;
        tree = new int[4 * n];
        build(arr, 1, 0, n - 1);
    }

    void build(int[] arr, int node, int l, int r) {
        if (l == r) { tree[node] = arr[l]; return; }
        int mid = (l + r) / 2;
        build(arr, 2 * node, l, mid);
        build(arr, 2 * node + 1, mid + 1, r);
        tree[node] = tree[2 * node] + tree[2 * node + 1];
    }

    void update(int node, int l, int r, int idx, int val) {
        if (l == r) { tree[node] = val; return; }
        int mid = (l + r) / 2;
        if (idx <= mid) update(2 * node, l, mid, idx, val);
        else update(2 * node + 1, mid + 1, r, idx, val);
        tree[node] = tree[2 * node] + tree[2 * node + 1];
    }

    void update(int idx, int val) { update(1, 0, n - 1, idx, val); }

    int query(int node, int l, int r, int ql, int qr) {
        if (qr < l || r < ql) return 0;
        if (ql <= l && r <= qr) return tree[node];
        int mid = (l + r) / 2;
        return query(2 * node, l, mid, ql, qr) +
               query(2 * node + 1, mid + 1, r, ql, qr);
    }

    int query(int ql, int qr) { return query(1, 0, n - 1, ql, qr); }
}

class SegTree:
    def __init__(self, arr):
        self.n = len(arr)
        self.tree = [0] * (4 * self.n)
        self._build(arr, 1, 0, self.n - 1)

    def _build(self, arr, node, l, r):
        if l == r:
            self.tree[node] = arr[l]
            return
        mid = (l + r) // 2
        self._build(arr, 2 * node, l, mid)
        self._build(arr, 2 * node + 1, mid + 1, r)
        self.tree[node] = self.tree[2 * node] + self.tree[2 * node + 1]

    def update(self, idx, val, node=1, l=0, r=None):
        if r is None: r = self.n - 1
        if l == r:
            self.tree[node] = val
            return
        mid = (l + r) // 2
        if idx <= mid:
            self.update(idx, val, 2 * node, l, mid)
        else:
            self.update(idx, val, 2 * node + 1, mid + 1, r)
        self.tree[node] = self.tree[2 * node] + self.tree[2 * node + 1]

    def query(self, ql, qr, node=1, l=0, r=None):
        if r is None: r = self.n - 1
        if qr < l or r < ql: return 0
        if ql <= l and r <= qr: return self.tree[node]
        mid = (l + r) // 2
        return (self.query(ql, qr, 2 * node, l, mid) +
                self.query(ql, qr, 2 * node + 1, mid + 1, r))

# Usage
st = SegTree([2, 1, 5, 3, 4, 7, 6, 8])
print(st.query(1, 5))  # 20
st.update(2, 9)
print(st.query(1, 5))  # 24

Key Points:
  • Any associative operation works with segment trees
  • The identity element is returned for "no overlap" cases
  • For non-commutative operations (like matrix multiplication), maintain left-to-right order
  • Memory: use 4n array size (not 2n) to handle non-power-of-2 sizes safely
