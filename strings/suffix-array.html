Suffix Array & LCP

A suffix array is the sorted order of all suffixes of a string — a compact, powerful index. Pair it with the LCP array (longest common prefix between adjacent sorted suffixes) and you can answer deep substring questions fast.

What They Are

   s = "banana"   (suffixes, then sorted)

   start  suffix          sorted →  SA   suffix       LCP
   0      banana                    5    a              -
   1      anana                     3    ana            1
   2      nana                      1    anana          3
   3      ana                       0    banana         0
   4      na                        4    na             0
   5      a                         2    nana           2

   SA  = [5, 3, 1, 0, 4, 2]   (start indices in sorted order)
   LCP = [-, 1, 3, 0, 0, 2]   (shared prefix with previous row)

Code: Suffix Array (O(n log²n)) + Kasai LCP

// O(n log^2 n) suffix array via prefix-doubling
vector<int> suffixArray(string s) {
    s += '$';
    int n = s.size();
    vector<int> sa(n), rank(n), tmp(n);
    for (int i = 0; i < n; i++) { sa[i] = i; rank[i] = s[i]; }
    for (int k = 1; k < n; k <<= 1) {
        auto cmp = [&](int a, int b) {
            if (rank[a] != rank[b]) return rank[a] < rank[b];
            int ra = a + k < n ? rank[a+k] : -1;
            int rb = b + k < n ? rank[b+k] : -1;
            return ra < rb;
        };
        sort(sa.begin(), sa.end(), cmp);
        tmp[sa[0]] = 0;
        for (int i = 1; i < n; i++)
            tmp[sa[i]] = tmp[sa[i-1]] + (cmp(sa[i-1], sa[i]) ? 1 : 0);
        rank = tmp;
    }
    return sa;
}

// Kasai's algorithm: LCP in O(n)
vector<int> kasai(const string& s, vector<int>& sa) {
    int n = s.size(), k = 0;
    vector<int> rank(n), lcp(n, 0);
    for (int i = 0; i < n; i++) rank[sa[i]] = i;
    for (int i = 0; i < n; i++) {
        if (rank[i] == n - 1) { k = 0; continue; }
        int j = sa[rank[i] + 1];
        while (i+k < n && j+k < n && s[i+k] == s[j+k]) k++;
        lcp[rank[i]] = k;
        if (k) k--;
    }
    return lcp;
}
// O(n log^2 n) suffix array
static Integer[] suffixArray(String s) {
    s += "$";
    int n = s.length();
    Integer[] sa = new Integer[n];
    int[] rank = new int[n], tmp = new int[n];
    for (int i = 0; i < n; i++) { sa[i] = i; rank[i] = s.charAt(i); }
    for (int k = 1; k < n; k <<= 1) {
        final int kk = k; final int[] r = rank;
        Comparator<Integer> cmp = (a, b) -> {
            if (r[a] != r[b]) return r[a] - r[b];
            int ra = a + kk < n ? r[a+kk] : -1;
            int rb = b + kk < n ? r[b+kk] : -1;
            return ra - rb;
        };
        Arrays.sort(sa, cmp);
        tmp[sa[0]] = 0;
        for (int i = 1; i < n; i++)
            tmp[sa[i]] = tmp[sa[i-1]] + (cmp.compare(sa[i-1], sa[i]) < 0 ? 1 : 0);
        System.arraycopy(tmp, 0, rank, 0, n);
    }
    return sa;
}
# Simple O(n log n) suffix array (good for moderate n)
def suffix_array(s):
    s += '$'
    n = len(s)
    sa = list(range(n))
    rank = [ord(c) for c in s]
    k = 1
    while k < n:
        sa.sort(key=lambda i: (rank[i], rank[i + k] if i + k < n else -1))
        tmp = [0] * n
        for i in range(1, n):
            prev, cur = sa[i-1], sa[i]
            key_prev = (rank[prev], rank[prev+k] if prev+k < n else -1)
            key_cur = (rank[cur], rank[cur+k] if cur+k < n else -1)
            tmp[cur] = tmp[prev] + (1 if key_cur > key_prev else 0)
        rank = tmp
        k <<= 1
    return sa

def kasai(s, sa):           # LCP in O(n)
    n = len(s); rank = [0]*n
    for i, p in enumerate(sa): rank[p] = i
    lcp = [0]*n; k = 0
    for i in range(n):
        if rank[i] == n-1: k = 0; continue
        j = sa[rank[i]+1]
        while i+k < n and j+k < n and s[i+k] == s[j+k]: k += 1
        lcp[rank[i]] = k
        if k: k -= 1
    return lcp

What It Unlocks

  • Substring search: binary search the SA in O(m log n).
  • Number of distinct substrings: n(n+1)/2 − Σ LCP.
  • Longest repeated substring: the maximum LCP value.
  • Longest common substring of two strings: concatenate with a separator, look at LCP across the boundary.
  • Faster builds exist: SA-IS in O(n); DC3 in O(n).

Common Pitfalls

1. Sentinel character: append a unique smallest char ($) so no suffix is a prefix of another.

2. LCP indexing: Kasai's lcp[rank[i]] is the LCP between SA[rank[i]] and the next — keep the convention straight.

3. Complexity: the simple sort-based build is O(n log²n); for n up to 10⁶ use SA-IS.

4. Tie-breaking: when ranks tie, compare the second half's rank — forgetting it corrupts the order.
