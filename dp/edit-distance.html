Edit Distance (Levenshtein)

Find the minimum number of operations (insert, delete, replace) to convert string A into string B. A fundamental string DP problem with wide applications.

Problem Formulation

   A = "horse"    B = "ros"

   horse → rorse  (replace 'h' with 'r')
   rorse → rose   (delete 'r')
   rose  → ros    (delete 'e')

   Edit distance = 3 operations

   Operations allowed:
   1. Insert a character
   2. Delete a character
   3. Replace a character

State & Transition

State: dp[i][j] = edit distance between A[0..i-1] and B[0..j-1]

Transition:
If A[i-1] == B[j-1]: dp[i][j] = dp[i-1][j-1] (characters match, no operation needed)
Else: dp[i][j] = 1 + min(
    dp[i-1][j-1],  // replace A[i-1] with B[j-1]
    dp[i][j-1],   // insert B[j-1] after A[i-1]
    dp[i-1][j]    // delete A[i-1]
)

Base: dp[i][0] = i (delete all of A), dp[0][j] = j (insert all of B) 
   How to think about the operations:

   dp[i-1][j-1] + 1  →  REPLACE: convert A[0..i-2] to B[0..j-2],
                         then replace A[i-1] with B[j-1]

   dp[i][j-1] + 1    →  INSERT: convert A[0..i-1] to B[0..j-2],
                         then insert B[j-1] at end

   dp[i-1][j] + 1    →  DELETE: convert A[0..i-2] to B[0..j-1],
                         then delete A[i-1] (it was extra)

// O(m*n) time, O(m*n) space
int editDistance(string& A, string& B) {
    int m = A.size(), n = B.size();
    vector<vector<int>> dp(m + 1, vector<int>(n + 1, 0));

    // Base cases
    for (int i = 0; i <= m; i++) dp[i][0] = i;
    for (int j = 0; j <= n; j++) dp[0][j] = j;

    for (int i = 1; i <= m; i++) {
        for (int j = 1; j <= n; j++) {
            if (A[i-1] == B[j-1]) {
                dp[i][j] = dp[i-1][j-1]; // match
            } else {
                dp[i][j] = 1 + min({
                    dp[i-1][j-1], // replace
                    dp[i][j-1],   // insert
                    dp[i-1][j]    // delete
                });
            }
        }
    }
    return dp[m][n];
}

// Space optimized: O(n) space
int editDistOpt(string& A, string& B) {
    int m = A.size(), n = B.size();
    vector<int> prev(n + 1), curr(n + 1);
    for (int j = 0; j <= n; j++) prev[j] = j;

    for (int i = 1; i <= m; i++) {
        curr[0] = i;
        for (int j = 1; j <= n; j++) {
            if (A[i-1] == B[j-1]) curr[j] = prev[j-1];
            else curr[j] = 1 + min({prev[j-1], curr[j-1], prev[j]});
        }
        swap(prev, curr);
    }
    return prev[n];
}

// O(m*n) time, O(m*n) space
static int editDistance(String A, String B) {
    int m = A.length(), n = B.length();
    int[][] dp = new int[m + 1][n + 1];

    for (int i = 0; i <= m; i++) dp[i][0] = i;
    for (int j = 0; j <= n; j++) dp[0][j] = j;

    for (int i = 1; i <= m; i++) {
        for (int j = 1; j <= n; j++) {
            if (A.charAt(i-1) == B.charAt(j-1)) {
                dp[i][j] = dp[i-1][j-1];
            } else {
                dp[i][j] = 1 + Math.min(
                    dp[i-1][j-1], // replace
                    Math.min(dp[i][j-1], dp[i-1][j]) // insert, delete
                );
            }
        }
    }
    return dp[m][n];
}

// Space optimized: O(n) space
static int editDistOpt(String A, String B) {
    int m = A.length(), n = B.length();
    int[] prev = new int[n + 1];
    int[] curr = new int[n + 1];
    for (int j = 0; j <= n; j++) prev[j] = j;

    for (int i = 1; i <= m; i++) {
        curr[0] = i;
        for (int j = 1; j <= n; j++) {
            if (A.charAt(i-1) == B.charAt(j-1))
                curr[j] = prev[j-1];
            else
                curr[j] = 1 + Math.min(prev[j-1],
                    Math.min(curr[j-1], prev[j]));
        }
        int[] temp = prev; prev = curr; curr = temp;
    }
    return prev[n];
}

# O(m*n) time, O(m*n) space
def edit_distance(A, B):
    m, n = len(A), len(B)
    dp = [[0] * (n + 1) for _ in range(m + 1)]

    for i in range(m + 1): dp[i][0] = i
    for j in range(n + 1): dp[0][j] = j

    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if A[i-1] == B[j-1]:
                dp[i][j] = dp[i-1][j-1]  # match
            else:
                dp[i][j] = 1 + min(
                    dp[i-1][j-1],  # replace
                    dp[i][j-1],    # insert
                    dp[i-1][j]     # delete
                )
    return dp[m][n]

# Space optimized: O(n) space
def edit_dist_opt(A, B):
    m, n = len(A), len(B)
    prev = list(range(n + 1))

    for i in range(1, m + 1):
        curr = [i] + [0] * n
        for j in range(1, n + 1):
            if A[i-1] == B[j-1]:
                curr[j] = prev[j-1]
            else:
                curr[j] = 1 + min(prev[j-1], curr[j-1], prev[j])
        prev = curr
    return prev[n]

Complexity

  • Time: O(m * n) where m = |A|, n = |B|
  • Space: O(m * n) for full table, O(min(m, n)) with space optimization
  • Note: Space-optimized version loses ability to reconstruct the sequence of operations (no backtracking)

Applications

   1. Spell Checking
      "teh" → "the" (distance = 1, swap)
      Suggest words within edit distance 1-2

   2. DNA Sequence Alignment
      GATTACA vs GCATGCU
      Used in bioinformatics (with weighted operations)

   3. Diff Tools (git diff)
      Find minimum edits between file versions

   4. Fuzzy String Matching
      Search engines: "amazn" matches "amazon"

   5. Natural Language Processing
      Autocorrect, OCR error correction

Variations

VariantDifference
Levenshtein DistanceInsert, delete, replace (cost 1 each)
Longest Common SubsequenceOnly insert and delete (no replace)
Damerau-LevenshteinAlso allows transposition of adjacent chars
Weighted Edit DistanceDifferent costs per operation or character pair
Hamming DistanceOnly replace (strings must be same length)
