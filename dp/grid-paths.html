Grid Paths & Min Path Sum

Navigate a grid from top-left to bottom-right. Count unique paths or find the minimum cost path. A visual introduction to 2D DP.

Problem 1: Unique Paths

In an m×n grid, you can only move right or down. Count the number of unique paths from (0,0) to (m-1,n-1).

State: dp[i][j] = number of ways to reach cell (i,j)

Transition: dp[i][j] = dp[i-1][j] + dp[i][j-1] (from above + from left)

Base: dp[0][j] = dp[i][0] = 1 (only one way to reach any cell in first row/column)

Problem 2: Min Path Sum

Each cell has a cost. Find path from top-left to bottom-right with minimum total cost.

// Unique Paths: O(m*n) time, O(m*n) space
int uniquePaths(int m, int n) {
    vector<vector<int>> dp(m, vector<int>(n, 1));

    for (int i = 1; i < m; i++) {
        for (int j = 1; j < n; j++) {
            dp[i][j] = dp[i-1][j] + dp[i][j-1];
        }
    }
    return dp[m-1][n-1];
}

// Space optimized: O(n)
int uniquePathsOpt(int m, int n) {
    vector<int> dp(n, 1);
    for (int i = 1; i < m; i++)
        for (int j = 1; j < n; j++)
            dp[j] += dp[j-1];
    return dp[n-1];
}

// Min Path Sum: O(m*n) time, O(1) extra space (modify grid)
int minPathSum(vector<vector<int>>& grid) {
    int m = grid.size(), n = grid[0].size();
    for (int i = 1; i < m; i++) grid[i][0] += grid[i-1][0];
    for (int j = 1; j < n; j++) grid[0][j] += grid[0][j-1];

    for (int i = 1; i < m; i++)
        for (int j = 1; j < n; j++)
            grid[i][j] += min(grid[i-1][j], grid[i][j-1]);

    return grid[m-1][n-1];
}

// Unique Paths: O(m*n) time
static int uniquePaths(int m, int n) {
    int[][] dp = new int[m][n];
    for (int i = 0; i < m; i++) dp[i][0] = 1;
    for (int j = 0; j < n; j++) dp[0][j] = 1;

    for (int i = 1; i < m; i++)
        for (int j = 1; j < n; j++)
            dp[i][j] = dp[i-1][j] + dp[i][j-1];

    return dp[m-1][n-1];
}

// Min Path Sum: O(m*n) time
static int minPathSum(int[][] grid) {
    int m = grid.length, n = grid[0].length;
    for (int i = 1; i < m; i++) grid[i][0] += grid[i-1][0];
    for (int j = 1; j < n; j++) grid[0][j] += grid[0][j-1];

    for (int i = 1; i < m; i++)
        for (int j = 1; j < n; j++)
            grid[i][j] += Math.min(grid[i-1][j], grid[i][j-1]);

    return grid[m-1][n-1];
}

# Unique Paths: O(m*n) time, O(n) space
def unique_paths(m, n):
    dp = [1] * n
    for i in range(1, m):
        for j in range(1, n):
            dp[j] += dp[j-1]
    return dp[n-1]

# Min Path Sum: O(m*n) time, O(1) extra
def min_path_sum(grid):
    m, n = len(grid), len(grid[0])
    for i in range(1, m):
        grid[i][0] += grid[i-1][0]
    for j in range(1, n):
        grid[0][j] += grid[0][j-1]

    for i in range(1, m):
        for j in range(1, n):
            grid[i][j] += min(grid[i-1][j], grid[i][j-1])

    return grid[m-1][n-1]

# With obstacles
def unique_paths_obstacles(grid):
    m, n = len(grid), len(grid[0])
    dp = [[0] * n for _ in range(m)]
    dp[0][0] = 1 if grid[0][0] == 0 else 0

    for i in range(m):
        for j in range(n):
            if grid[i][j] == 1:
                dp[i][j] = 0
            else:
                if i > 0: dp[i][j] += dp[i-1][j]
                if j > 0: dp[i][j] += dp[i][j-1]
    return dp[m-1][n-1]

Complexity

  • Unique Paths: O(m*n) time, O(n) space optimized
  • Min Path Sum: O(m*n) time, O(1) extra if modifying grid in-place
  • With Obstacles: Same complexity, just set dp=0 at obstacle cells
  • Math formula: C(m+n-2, m-1) for unique paths without obstacles
